### 2.2 工具执行中的权限检查

**位置**: `src/QueryEngine.ts` (推断)

```typescript
private async executeToolCalls(
  toolCalls: ToolUseBlock[],
  canUseTool: CanUseToolFn,
  tools: Tools
): Promise<ToolResult[]> {
  const results: ToolResult[] = []
  
  for (const toolCall of toolCalls) {
    // 1. 查找工具
    const tool = tools.find(t => t.name === toolCall.name)
    
    if (!tool) {
      results.push(createErrorResult(toolCall.id, `Unknown tool: ${toolCall.name}`))
      continue
    }
    
    // 2. 验证输入
    const validation = tool.validateInput(toolCall.input)
    
    if (!validation.valid) {
      results.push(createErrorResult(toolCall.id, `Invalid input: ${validation.errors.map(e => e.message).join(', ')}`))
      continue
    }
    
    // 3. === 权限检查 ===
    const permissionResult = await canUseTool(
      tool,
      validation.input,
      this.getToolContext(),
      this.currentAssistantMessage,
      toolCall.id
    )
    
    if (permissionResult.behavior === 'deny') {
      results.push(createErrorResult(toolCall.id, `Permission denied: ${permissionResult.reason}`))
      continue
    }
    
    // 4. 执行工具
    try {
      const toolResult = await tool.execute(validation.input, this.getToolContext())
      
      results.push({
        type: 'tool_result',
        tool_use_id: toolCall.id,
        content: toolResult.content,
        is_error: toolResult.success === false,
      })
    } catch (error) {
      results.push(createErrorResult(toolCall.id, `Error: ${errorMessage(error)}`))
    }
  }
  
  return results
}
```

### 4.2 审批决策处理

```typescript
// src/hooks/useApprovalHistory.js (推断)

async function handlePermissionDecision(
  toolName: string,
  input: unknown,
  decision: 'approve' | 'deny' | 'always_allow'
): Promise<void> {
  switch (decision) {
    case 'approve':
      // 仅此一次批准
      await recordApproval(toolName, 'once')
      break
      
    case 'always_allow':
      // 总是允许，记录到审批历史
      await recordApproval(toolName, 'always')
      break
      
    case 'deny':
      // 拒绝，记录到拒绝历史
      await recordDenial(toolName, input)
      break
  }
}

async function recordApproval(toolName: string, type: 'once' | 'always'): Promise<void> {
  const history = await loadApprovalHistory()
  
  if (!history[toolName]) {
    history[toolName] = { toolName, decisions: [], alwaysAllow: false }
  }
  
  if (type === 'always') {
    history[toolName].alwaysAllow = true
  }
  
  history[toolName].decisions.push({
    timestamp: Date.now(),
    decision: 'approve',
    type,
  })
  
  await saveApprovalHistory(history)
}

async function recordDenial(toolName: string, input: unknown): Promise<void> {
  const history = await loadApprovalHistory()
  
  if (!history[toolName]) {
    history[toolName] = { toolName, decisions: [], alwaysDeny: false }
  }
  
  history[toolName].decisions.push({
    timestamp: Date.now(),
    decision: 'deny',
    input,
  })
  
  await saveApprovalHistory(history)
}
```

```typescript
// src/hooks/useApprovalHistory.js (推断)

export function useApprovalHistory() {
  const [approvalHistory, setApprovalHistory] = useState<ApprovalHistory>({})
  
  // 加载审批历史
  useEffect(() => {
    loadApprovalHistory().then(setApprovalHistory)
  }, [])
  
  // 检查工具是否被总是允许
  const isAlwaysAllowed = useCallback((toolName: string): boolean => {
    return approvalHistory[toolName]?.alwaysAllow ?? false
  }, [approvalHistory])
  
  // 检查工具是否被总是拒绝
  const isAlwaysDenied = useCallback((toolName: string): boolean => {
    return approvalHistory[toolName]?.alwaysDeny ?? false
  }, [approvalHistory])
  
  // 清除工具的审批历史
  const clearHistory = useCallback((toolName: string) => {
    setApprovalHistory(prev => {
      const updated = { ...prev }
      delete updated[toolName]
      return updated
    })
  }, [])
  
  return {
    approvalHistory,
    isAlwaysAllowed,
    isAlwaysDenied,
    clearHistory,
  }
}
```

```typescript
// 文件系统安全检查

function isPathAllowed(filePath: string, allowedPaths: string[]): boolean {
  const resolvedPath = path.resolve(filePath)
  
  // 检查是否在允许的路径内
  for (const allowedPath of allowedPaths) {
    const resolvedAllowed = path.resolve(allowedPath)
    if (resolvedPath.startsWith(resolvedAllowed)) {
      return true
    }
  }
  
  return false
}

function isPathBlocked(filePath: string, blockedPaths: string[]): boolean {
  const resolvedPath = path.resolve(filePath)
  
  for (const blockedPath of blockedPaths) {
    const resolvedBlocked = path.resolve(blockedPath)
    if (resolvedPath.startsWith(resolvedBlocked)) {
      return true
    }
  }
  
  return false
}

// 敏感路径检查
const SENSITIVE_PATHS = [
  '/etc',
  '/proc',
  '/sys',
  '/root',
  'C:\\Windows\\System32',
  'C:\\Users\\Administrator',
]

function isSensitivePath(filePath: string): boolean {
  return SENSITIVE_PATHS.some(p => filePath.startsWith(p))
}
```

### 6.2 命令执行边界

**位置**: `src/utils/permissions/commands.js` (推断)

```typescript
// 命令安全检查

const BLOCKED_COMMAND_PATTERNS = [
  /^rm\s+(-[rf]+\s+)?\//,           // rm -rf /
  /^rm\s+(-[rf]+\s+)?~/,            // rm -rf ~
  /^mkfs/,                          // 格式化
  /^dd\s+of=/,                      // dd 写入
  /^chmod\s+(-R\s+)?777/,           // chmod 777
  /^chown\s+(-R\s+)?[0-9]+:[0-9]+/, // chown root
  /^: fork\(\)\s*\{/,               // fork bomb
  /curl.*\|\s*(ba)?sh/,             // curl | sh
  /wget.*\|\s*(ba)?sh/,             // wget | sh
  /nc\s+.*-e\s+(ba)?sh/,            // nc -e sh (reverse shell)
]

function isCommandBlocked(command: string): boolean {
  for (const pattern of BLOCKED_COMMAND_PATTERNS) {
    if (pattern.test(command)) {
      return true
    }
  }
  
  return false
}

// 命令风险级别评估
function assessCommandRisk(command: string): 'low' | 'medium' | 'high' {
  // 只读命令：低风险
  const readOnlyCommands = ['ls', 'cat', 'head', 'tail', 'grep', 'find', 'stat']
  if (readOnlyCommands.some(cmd => command.startsWith(cmd))) {
    return 'low'
  }
  
  // 创建/修改命令：中风险
  const modifyCommands = ['mkdir', 'touch', 'cp', 'mv', 'ln']
  if (modifyCommands.some(cmd => command.startsWith(cmd))) {
    return 'medium'
  }
  
  // 删除/执行命令：高风险
  const highRiskCommands = ['rm', 'bash', 'sh', 'chmod', 'chown', 'sudo']
  if (highRiskCommands.some(cmd => command.startsWith(cmd))) {
    return 'high'
  }
  
  return 'medium'
}
```

### 6.3 网络访问边界

**位置**: `src/utils/permissions/network.js` (推断)

```typescript
// 网络访问检查

const BLOCKED_HOSTS = [
  'localhost',
  '127.0.0.1',
  '169.254.169.254',  // AWS metadata
  'metadata.google.internal',  // GCP metadata
  '100.100.100.100',  // Alibaba metadata
]

const ALLOWED_PROTOCOLS = ['http:', 'https:']

function isUrlAllowed(url: string, allowedHosts?: string[]): boolean {
  const parsed = new URL(url)
  
  // 检查协议
  if (!ALLOWED_PROTOCOLS.includes(parsed.protocol)) {
    return false
  }
  
  // 检查阻止的主机
  if (BLOCKED_HOSTS.some(host => 
    parsed.hostname === host || parsed.hostname.endsWith(`.${host}`)
  )) {
    return false
  }
  
  // 如果有允许列表，检查是否在允许列表中
  if (allowedHosts && allowedHosts.length > 0) {
    return allowedHosts.some(host => 
      parsed.hostname === host || parsed.hostname.endsWith(`.${host}`)
    )
  }
  
  return true
}
```

## 7. 典型权限场景

### 7.1 场景：读取文件 (低风险)

```
用户："帮我读取 config.json 文件"

LLM 决策：
└─> 调用 FileReadTool { file_path: "config.json" }

权限检查流程：
1. FileReadTool.getPermission() 返回:
   { level: 'low', requiresApproval: false }

2. 检查权限模式:
   - default 模式：需要审批
   - auto 模式：自动允许 ✓

3. 检查审批历史:
   - 无总是允许/拒绝记录

4. 最终决策:
   - auto 模式 + low 风险 = 自动允许

结果：工具自动执行，无需用户审批
```

### 7.2 场景：编辑文件 (中风险)

```
用户："帮我把这个函数的返回值改为 true"

LLM 决策：
└─> 调用 FileEditTool { file_path: "src/utils.js", old_string: "...", new_string: "..." }

权限检查流程：
1. FileEditTool.getPermission() 返回:
   { level: 'medium', requiresApproval: true }

2. 检查权限模式:
   - default 模式：需要审批
   - auto 模式：需要审批

3. 检查审批历史:
   - 无总是允许/拒绝记录

4. 显示审批对话框:
   ┌─────────────────────────────────────────────────────────────┐
   │ Permission Required                                         │
   ├─────────────────────────────────────────────────────────────┤
   │ Tool: FileEdit                                              │
   │ Edit file: src/utils.js                                     │
   │                                                             │
   │ Changes:                                                    │
   │ - return false                                              │
   │ + return true                                               │
   │                                                             │
   │ [Y] Approve  |  [N] Deny  |  [A] Always allow              │
   └─────────────────────────────────────────────────────────────┘

5. 用户决策:
   - 用户按 Y 批准

结果：工具执行，仅本次允许
```

### 7.4 场景：危险命令被阻止

```
用户："运行 rm -rf /"

LLM 决策（如果 LLM 决定执行这个危险命令）:
└─> 调用 BashTool { command: "rm -rf /" }

权限检查流程：
1. BashTool.getPermission() 检查命令:
   检测到危险命令 "rm -rf /"
   返回：{ level: 'high', requiresApproval: true, blocked: true }

2. 命令被全局禁止:
   直接拒绝，不显示审批对话框

结果：
┌─────────────────────────────────────────────────────────────┐
│ Error: Command "rm -rf /" is blocked for safety reasons.    │
└─────────────────────────────────────────────────────────────┘
```

### 7.5 场景：总是允许某个工具

```
用户频繁使用 FileReadTool

2. 用户按 [A] Always allow

审批历史记录：
{
  "FileRead": {
    toolName: "FileRead",

  }

后续调用：

2. 发现 alwaysAllow = true
3. 自动允许，无需审批

结果：FileReadTool 在此会话中自动执行
```

## 总结

权限审批流程是 Claude Code 安全设计的核心：

1. **多层次检查**: 权限模式→审批历史→用户审批→边界检查
2. **风险分级**: low/medium/high 三级风险分类
3. **灵活模式**: 5 种权限模式适应不同场景
4. **用户控制**: 用户可决定单次批准或总是允许
5. **安全边界**: 文件系统、命令、网络的多重边界检查

这个系统确保了 AI 工具在提供强大功能的同时，不会对系统造成意外损害。

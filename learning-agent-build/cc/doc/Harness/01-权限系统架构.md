## 1. 权限检查入口

### 1.1 useCanUseTool Hook

**源码**: `src/hooks/useCanUseTool.tsx`

`useCanUseTool` 是权限检查的统一入口，所有工具调用都必须通过这个 Hook 进行权限验证。

**函数签名**:

```typescript
export type CanUseToolFn<Input extends Record<string, unknown>> = (
  tool: Tool,
  input: Input,
  toolUseContext: ToolUseContext,
  assistantMessage: AssistantMessage,
  toolUseID: string,
  forceDecision?: PermissionDecision<Input>,
) => Promise<PermissionDecision<Input>>
```

**核心流程**:

```typescript
async (tool, input, toolUseContext, assistantMessage, toolUseID, forceDecision) => {
  // ========== 创建权限上下文 ==========
  const ctx = createPermissionContext(
    tool,
    input,
    assistantMessage,
    toolUseID,
    toolUseContext,
  )
  
  // ========== 获取权限决策 ==========
  const decisionPromise = forceDecision !== undefined
    ? Promise.resolve(forceDecision)
    : hasPermissionsToUseTool(tool, input, toolUseContext, assistantMessage, toolUseID)
  
  // ========== 处理决策结果 ==========
  return decisionPromise.then(async result => {
    if (result.behavior === "allow") {
      // 直接允许
      resolve(ctx.buildAllow(result.updatedInput ?? input, {
        decisionReason: result.decisionReason
      }))
    }
    
    switch (result.behavior) {
      case "deny":
        // 拒绝并记录
        resolve(result)
        break
      case "ask":
        // 进入 UI 对话流程
        handleInteractivePermission({...}, resolve)
        break
    }
  })
}
```

### 2.1 hasPermissionsToUseTool

**源码**: `src/utils/permissions/permissions.ts:473-956`

这是权限决策的核心函数，执行完整的权限检查流程。

**主流程结构**:

```typescript
export const hasPermissionsToUseTool: CanUseToolFn = async (
  tool,
  input,
  context,
  assistantMessage,
  toolUseID,
): Promise<PermissionDecision> => {
  const result = await hasPermissionsToUseToolInner(tool, input, context)
  
  // ========== 成功时重置拒绝计数 ==========
  if (result.behavior === 'allow') {
    // 在 auto 模式下，任何允许的操作都会重置连续拒绝计数
    if (feature('TRANSCRIPT_CLASSIFIER')) {
      const currentDenialState = context.localDenialTracking ?? context.getAppState().denialTracking
      if (context.getAppState().toolPermissionContext.mode === 'auto' &&
          currentDenialState && currentDenialState.consecutiveDenials > 0) {
        const newDenialState = recordSuccess(currentDenialState)
        persistDenialState(context, newDenialState)
      }
    }
    return result
  }
  
  // ========== dontAsk 模式转换 ==========
  if (result.behavior === 'ask') {
    const appState = context.getAppState()
    
    if (appState.toolPermissionContext.mode === 'dontAsk') {
      return {
        behavior: 'deny',
        decisionReason: { type: 'mode', mode: 'dontAsk' },
        message: DONT_ASK_REJECT_MESSAGE(tool.name),
      }
    }
    
    // ========== auto 模式：使用 AI 分类器 ==========
    if (feature('TRANSCRIPT_CLASSIFIER') &&
        (appState.toolPermissionContext.mode === 'auto' ||
         appState.toolPermissionContext.mode === 'plan')) {
      // ... auto 模式分类器逻辑
    }
    
    // ========== 无交互能力时自动拒绝 ==========
    if (appState.toolPermissionContext.shouldAvoidPermissionPrompts) {
      const hookDecision = await runPermissionRequestHooksForHeadlessAgent(...)
      if (hookDecision) return hookDecision
      return {
        behavior: 'deny',
        decisionReason: { type: 'asyncAgent', reason: '...' },
        message: AUTO_REJECT_MESSAGE(tool.name),
      }
    }
  }
  
  return result
}
```

### 2.2 hasPermissionsToUseToolInner

**源码**: `src/utils/permissions/permissions.ts:1158-1319`

执行实际的规则匹配和工具检查，不涉及模式转换。

**检查步骤**:

```typescript
async function hasPermissionsToUseToolInner(
  tool: Tool,
  input: { [key: string]: unknown },
  context: ToolUseContext,
): Promise<PermissionDecision> {
  const appState = context.getAppState()
  
  // ========== 步骤 1a: 工具级 deny 规则 ==========
  const denyRule = getDenyRuleForTool(appState.toolPermissionContext, tool)
  if (denyRule) {
    return {
      behavior: 'deny',
      decisionReason: { type: 'rule', rule: denyRule },
      message: `Permission to use ${tool.name} has been denied.`,
    }
  }
  
  // ========== 步骤 1b: 工具级 ask 规则 ==========
  const askRule = getAskRuleForTool(appState.toolPermissionContext, tool)
  if (askRule) {
    const canSandboxAutoAllow = tool.name === BASH_TOOL_NAME &&
      SandboxManager.isSandboxingEnabled() &&
      SandboxManager.isAutoAllowBashIfSandboxedEnabled() &&
      shouldUseSandbox(input)
    
    if (!canSandboxAutoAllow) {
      return {
        behavior: 'ask',
        decisionReason: { type: 'rule', rule: askRule },
        message: createPermissionRequestMessage(tool.name),
      }
    }
  }
  
  // ========== 步骤 1c: 工具特定权限检查 ==========
  let toolPermissionResult: PermissionResult = {
    behavior: 'passthrough',
    message: createPermissionRequestMessage(tool.name),
  }
  try {
    const parsedInput = tool.inputSchema.parse(input)
    toolPermissionResult = await tool.checkPermissions(parsedInput, context)
  } catch (e) {
    if (e instanceof AbortError || e instanceof APIUserAbortError) throw e
    logError(e)
  }
  
  // ========== 步骤 1d: 工具实现拒绝 ==========
  if (toolPermissionResult?.behavior === 'deny') {
    return toolPermissionResult
  }
  
  // ========== 步骤 1e: 需要用户交互的工具 ==========
  if (tool.requiresUserInteraction?.() && toolPermissionResult?.behavior === 'ask') {
    return toolPermissionResult
  }
  
  // ========== 步骤 1f: 内容级 ask 规则 ==========
  if (toolPermissionResult?.behavior === 'ask' &&
      toolPermissionResult.decisionReason?.type === 'rule' &&
      toolPermissionResult.decisionReason.rule.ruleBehavior === 'ask') {
    return toolPermissionResult
  }
  
  // ========== 步骤 1g: 安全检查 ==========
  if (toolPermissionResult?.behavior === 'ask' &&
      toolPermissionResult.decisionReason?.type === 'safetyCheck') {
    return toolPermissionResult
  }
  
  // ========== 步骤 2a: bypassPermissions 模式 ==========
  appState = context.getAppState()
  const shouldBypassPermissions =
    appState.toolPermissionContext.mode === 'bypassPermissions' ||
    (appState.toolPermissionContext.mode === 'plan' &&
     appState.toolPermissionContext.isBypassPermissionsModeAvailable)
  if (shouldBypassPermissions) {
    return {
      behavior: 'allow',
      updatedInput: getUpdatedInputOrFallback(toolPermissionResult, input),
      decisionReason: { type: 'mode', mode: appState.toolPermissionContext.mode },
    }
  }
  
  // ========== 步骤 2b: 工具级 allow 规则 ==========
  const alwaysAllowedRule = toolAlwaysAllowedRule(appState.toolPermissionContext, tool)
  if (alwaysAllowedRule) {
    return {
      behavior: 'allow',
      updatedInput: getUpdatedInputOrFallback(toolPermissionResult, input),
      decisionReason: { type: 'rule', rule: alwaysAllowedRule },
    }
  }
  
  // ========== 步骤 3: passthrough 转 ask ==========
  const result: PermissionDecision =
    toolPermissionResult.behavior === 'passthrough'
      ? {
          ...toolPermissionResult,
          behavior: 'ask',
          message: createPermissionRequestMessage(
            tool.name,
            toolPermissionResult.decisionReason,
          ),
        }
      : toolPermissionResult
  
  return result
}
```

export function getAllowRules(context: ToolPermissionContext): PermissionRule[] {
  return PERMISSION_RULE_SOURCES.flatMap(source =>

### 3.3 工具匹配

```typescript
// 检查整个工具是否匹配规则（不考虑内容）
function toolMatchesRule(
  tool: Pick<Tool, 'name' | 'mcpInfo'>,
  rule: PermissionRule,
): boolean {
  // 规则必须没有 content 才能匹配整个工具
  if (rule.ruleValue.ruleContent !== undefined) return false
  
  const nameForRuleMatch = getToolNameForPermissionCheck(tool)
  
  // 直接工具名匹配
  if (rule.ruleValue.toolName === nameForRuleMatch) return true
  
  // MCP 服务器级权限：规则 "mcp__server1" 匹配 "mcp__server1__tool1"
  const ruleInfo = mcpInfoFromString(rule.ruleValue.toolName)
  const toolInfo = mcpInfoFromString(nameForRuleMatch)
  
  return (
    ruleInfo !== null &&
    toolInfo !== null &&
    (ruleInfo.toolName === undefined || ruleInfo.toolName === '*') &&
    ruleInfo.serverName === toolInfo.serverName
  )
}

// 检查工具是否有 deny 规则
export function getDenyRuleForTool(
  context: ToolPermissionContext,
  tool: Pick<Tool, 'name' | 'mcpInfo'>,
): PermissionRule | null {
  return getDenyRules(context).find(rule => toolMatchesRule(tool, rule)) || null
}
```

### 3.4 按内容获取规则

```typescript
// 获取工具的内容级规则（如 Bash(npm install:*)）
export function getRuleByContentsForTool(
  context: ToolPermissionContext,
  tool: Tool,
  behavior: PermissionBehavior,
): Map<string, PermissionRule> {
  return getRuleByContentsForToolName(
    context,
    getToolNameForPermissionCheck(tool),
    behavior,
  )
}

export function getRuleByContentsForToolName(
  context: ToolPermissionContext,
  toolName: string,
  behavior: PermissionBehavior,
): Map<string, PermissionRule> {
  const ruleByContents = new Map<string, PermissionRule>()
  let rules: PermissionRule[] = []
  
  switch (behavior) {
    case 'allow': rules = getAllowRules(context); break
    case 'deny': rules = getDenyRules(context); break
    case 'ask': rules = getAskRules(context); break
  }
  
  for (const rule of rules) {
    if (rule.ruleValue.toolName === toolName &&
        rule.ruleValue.ruleContent !== undefined &&
        rule.ruleBehavior === behavior) {
      ruleByContents.set(rule.ruleValue.ruleContent, rule)
    }
  }
  return ruleByContents
}
```

## 6. 关键设计特点

### 6.1 多层次检查

1. **工具级规则**：快速路径，匹配整个工具（如 `Bash`）
2. **内容级规则**：精确匹配命令内容（如 `Bash(npm install:*)`）
3. **工具特定检查**：每个工具可以实现自己的 `checkPermissions`
4. **安全检查**：路径约束、命令注入检测等通用安全检查

### 6.2 模式优先级

```
bypassPermissions > dontAsk > auto > 正常询问
```

### 6.3 拒绝追踪

Auto 模式下使用拒绝追踪来决定何时回退到用户询问：

## 7. 总结

权限系统是整个 Harness 工程的核心，具有以下特点：

1. **统一入口**：所有工具调用都通过 `useCanUseTool` Hook
2. **分层检查**：规则 → 工具检查 → 安全检查 → 模式转换
3. **灵活配置**：多来源规则、多模式支持
4. **可解释性**：每个决策都记录原因
5. **渐进式自动化**：从询问用户 → Hook 自动化 → AI 分类器自动化

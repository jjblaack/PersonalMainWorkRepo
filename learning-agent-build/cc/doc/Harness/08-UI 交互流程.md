# Harness 工程专题 - 08-UI 交互流程

UI 交互流程是用户与 Harness 系统交互的界面，包括权限弹窗、Bridge 通信、Channel 响应等。

## 1. UI 队列管理

```typescript
type ConfirmQueueItem = {
  assistantMessage: AssistantMessage
  tool: Tool
  description: string
  input: Record<string, unknown>
  toolUseContext: ToolUseContext
  toolUseID: string
  permissionResult: PermissionDecision
  permissionPromptStartTimeMs: number
  
  // 分类器状态（可选）
  classifierCheckInProgress?: boolean
  classifierAutoApproved?: boolean
  classifierMatchedRule?: string
  
  // 回调
  onAbort: () => void
  onAllow: (
    updatedInput: Record<string, unknown>,
    permissionUpdates: PermissionUpdate[],
    feedback?: string,
    contentBlocks?: ContentBlockParam[],
  ) => Promise<void>
  onReject: (
    feedback?: string,
    contentBlocks?: ContentBlockParam[],
  ) => void
  onUserInteraction: () => void
  onDismissCheckmark: () => void
  recheckPermission: () => Promise<void>
}
```

### 1.2 队列操作

```typescript
class PermissionContext {
  private confirmQueue: ConfirmQueueItem[] = []
  
  pushToQueue(item: ConfirmQueueItem): void {
    this.confirmQueue.push(item)
    // 触发 UI 更新
  }
  
  removeFromQueue(): void {
    this.confirmQueue.shift()
    // 触发 UI 更新
  }
  
  updateQueueItem(updates: Partial<ConfirmQueueItem>): void {
    if (this.confirmQueue.length > 0) {
      Object.assign(this.confirmQueue[0], updates)
    }
  }
}
```

## 2. 交互处理器

### 2.1 handleInteractivePermission

**源码**: `src/hooks/toolPermission/handlers/interactiveHandler.ts`

这是主交互处理器，处理本地用户的权限请求。

```typescript
function handleInteractivePermission(
  params: InteractivePermissionParams,
  resolve: (decision: PermissionDecision) => void,
): void {
  const {
    ctx,
    description,
    result,
    awaitAutomatedChecksBeforeDialog,
    bridgeCallbacks,
    channelCallbacks,
  } = params
  
  // ========== 原子检查工具 ==========
  const { resolve: resolveOnce, isResolved, claim } = createResolveOnce(resolve)
  let userInteracted = false
  
  const permissionPromptStartTimeMs = Date.now()
  const displayInput = result.updatedInput ?? ctx.input
  
  // ========== 推送到队列 ==========
  ctx.pushToQueue({
    assistantMessage: ctx.assistantMessage,
    tool: ctx.tool,
    description,
    input: displayInput,
    toolUseContext: ctx.toolUseContext,
    toolUseID: ctx.toolUseID,
    permissionResult: result,
    permissionPromptStartTimeMs,
    
    // 分类器状态
    ...(feature('BASH_CLASSIFIER')
      ? {
          classifierCheckInProgress:
            !!result.pendingClassifierCheck &&
            !awaitAutomatedChecksBeforeDialog,
        }
      : {}),
    
    // 用户开始交互
    onUserInteraction() {
      const GRACE_PERIOD_MS = 200
      if (Date.now() - permissionPromptStartTimeMs < GRACE_PERIOD_MS) return
      userInteracted = true
      clearClassifierChecking(ctx.toolUseID)
    },
    
    //  dismissal 勾选标记
    onDismissCheckmark() {
      if (checkmarkTransitionTimer) {
        clearTimeout(checkmarkTransitionTimer)
        ctx.removeFromQueue()
      }
    },
    
    // 中止
    onAbort() {
      if (!claim()) return
      if (bridgeCallbacks && bridgeRequestId) {
        bridgeCallbacks.sendResponse(bridgeRequestId, {
          behavior: 'deny',
          message: 'User aborted',
        })
        bridgeCallbacks.cancelRequest(bridgeRequestId)
      }
      channelUnsubscribe?.()
      ctx.logCancelled()
      ctx.logDecision({ decision: 'reject', source: { type: 'user_abort' } })
      resolveOnce(ctx.cancelAndAbort(undefined, true))
    },
    
    // 允许
    async onAllow(
      updatedInput,
      permissionUpdates: PermissionUpdate[],
      feedback?: string,
      contentBlocks?: ContentBlockParam[],
    ) {
      if (!claim()) return
      
      if (bridgeCallbacks && bridgeRequestId) {
        bridgeCallbacks.sendResponse(bridgeRequestId, {
          behavior: 'allow',
          updatedInput,
          updatedPermissions: permissionUpdates,
        })
        bridgeCallbacks.cancelRequest(bridgeRequestId)
      }
      channelUnsubscribe?.()
      
      resolveOnce(
        await ctx.handleUserAllow(
          updatedInput,
          permissionUpdates,
          feedback,
          permissionPromptStartTimeMs,
          contentBlocks,
          result.decisionReason,
        ),
      )
    },
    
    // 拒绝
    onReject(feedback?: string, contentBlocks?: ContentBlockParam[]) {
      if (!claim()) return
      
      if (bridgeCallbacks && bridgeRequestId) {
        bridgeCallbacks.sendResponse(bridgeRequestId, {
          behavior: 'deny',
          message: feedback ?? 'User denied permission',
        })
        bridgeCallbacks.cancelRequest(bridgeRequestId)
      }
      channelUnsubscribe?.()
      
      ctx.logDecision(
        {
          decision: 'reject',
          source: { type: 'user_reject', hasFeedback: !!feedback },
        },
        { permissionPromptStartTimeMs },
      )
      resolveOnce(ctx.cancelAndAbort(feedback, undefined, contentBlocks))
    },
    
    // 重新检查权限（CCR 模式切换）
    async recheckPermission() {
      if (isResolved()) return
      
      const freshResult = await hasPermissionsToUseTool(
        ctx.tool,
        ctx.input,
        ctx.toolUseContext,
        ctx.assistantMessage,
        ctx.toolUseID,
      )
      
      if (freshResult.behavior === 'allow') {
        if (!claim()) return
        if (bridgeCallbacks && bridgeRequestId) {
          bridgeCallbacks.cancelRequest(bridgeRequestId)
        }
        channelUnsubscribe?.()
        ctx.removeFromQueue()
        ctx.logDecision({ decision: 'accept', source: 'config' })
        resolveOnce(ctx.buildAllow(freshResult.updatedInput ?? ctx.input))
      }
    },
  })
  
  // ========== 异步执行 Hook ==========
  if (!awaitAutomatedChecksBeforeDialog) {
    void (async () => {
      if (isResolved()) return
      
      const hookDecision = await ctx.runHooks(...)
      if (!hookDecision || !claim()) return
      
      if (bridgeCallbacks && bridgeRequestId) {
        bridgeCallbacks.cancelRequest(bridgeRequestId)
      }
      channelUnsubscribe?.()
      ctx.removeFromQueue()
      resolveOnce(hookDecision)
    })()
  }
  
  // ========== 异步执行分类器 ==========
  if (
    feature('BASH_CLASSIFIER') &&
    result.pendingClassifierCheck &&
    ctx.tool.name === BASH_TOOL_NAME &&
    !awaitAutomatedChecksBeforeDialog
  ) {
    setClassifierChecking(ctx.toolUseID)
    void executeAsyncClassifierCheck(
      result.pendingClassifierCheck,
      ctx.toolUseContext.abortController.signal,
      ctx.toolUseContext.options.isNonInteractiveSession,
      {
        shouldContinue: () => !isResolved() && !userInteracted,
        onComplete: () => {
          clearClassifierChecking(ctx.toolUseID)
        },
        onAllow: decisionReason => {
          if (!claim()) return
          
          // 显示自动审批过渡
          if (feature('TRANSCRIPT_CLASSIFIER')) {
            ctx.updateQueueItem({
              classifierCheckInProgress: false,
              classifierAutoApproved: true,
              classifierMatchedRule: matchedRule,
            })
            
            // 保持勾选标记 3 秒/1 秒
            const checkmarkMs = getTerminalFocused() ? 3000 : 1000
            checkmarkTransitionTimer = setTimeout(() => {
              ctx.removeFromQueue()
            }, checkmarkMs)
          }
          
          ctx.logDecision({ decision: 'accept', source: { type: 'classifier' } })
          resolveOnce(ctx.buildAllow(ctx.input, { decisionReason }))
        },
      },
    )
  }
}
```

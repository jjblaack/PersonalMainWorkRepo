# 工人模块设计

> 工人 (Worker) 是超级 Agent 的执行单元，负责具体任务的执行。

---

## 1. 工人的定位和职责

### 1.1 工人是什么

```
┌─────────────────────────────────────────────────────────┐
│                    工人 (Worker)                          │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  定义：实际执行任务的 Agent 实例                            │
│                                                         │
│  本质：                                                 │
│  - 一个 LLM 会话 (带有特定上下文)                          │
│  - 一组可用工具 (Read, Write, Bash, etc.)                │
│  - 一个执行循环 (接收指令→执行→报告)                     │
│                                                         │
│  类比：                                                 │
│  - 管家 = 项目经理，工人 = 工程师                       │
│  - 管家 = 指挥家，工人 = 乐手                           │
│                                                         │
│  工人不負責：                                           │
│  - 任务分解 (由管家做)                                  │
│  - 工人调度 (由管家做)                                  │
│  - 结果验收 (由管家做)                                  │
│                                                         │
│  工人负责：                                             │
│  - 理解分配的任务                                       │
│  - 使用工具执行任务                                     │
│  - 报告进度和结果                                       │
│                                                         │
└─────────────────────────────────────────────────────────┘
```

### 1.2 工人类型

```typescript
// 工人类型
type WorkerType = 
  | 'research'        // 研究类：信息收集、代码探索
  | 'implementation'  // 实现类：编码、修改
  | 'test'            // 测试类：编写和运行测试
  | 'documentation'   // 文档类：编写文档、注释
  | 'verification'    // 验证类：审查、验证
  | 'general'         // 通用类：可以执行多种任务

// 不同类型工人的工具配置
const WORKER_TOOL_CONFIGS: Record<WorkerType, string[]> = {
  research: ['Read', 'Grep', 'Glob', 'Bash'],
  implementation: ['Read', 'Write', 'Edit', 'Bash'],
  test: ['Read', 'Write', 'Bash'],
  documentation: ['Read', 'Write'],
  verification: ['Read', 'Bash'],
  general: ['Read', 'Write', 'Edit', 'Grep', 'Glob', 'Bash'],
}
```

## 2. 工人接口定义

### 2.1 核心接口

```typescript
// 工人状态
type WorkerStatus = 
  | 'idle'         // 空闲，可以接受任务
  | 'busy'         // 忙碌，正在执行任务
  | 'paused'       // 暂停，等待依赖或指令
  | 'error'        // 错误，需要干预

// 工人接口
interface Worker {
  // 基本信息
  readonly id: string
  readonly type: WorkerType
  status: WorkerStatus
  currentTask?: string  // 当前任务 ID
  
  // 任务分配
  assign(spec: TaskSpec): Promise<void>
  continue(message: string): Promise<void>
  stop(): Promise<void>
  
  // 状态查询
  getStatus(): WorkerStatus
  getOutput(): string
  getActivities(): Activity[]
  
  // 生命周期
  cleanup(): Promise<void>
}

// 任务规格
interface TaskSpec {
  subject: string         // 任务主题
  description: string     // 任务描述
  context?: string        // 上下文/背景
  outputFormat?: string   // 期望的输出格式
  constraints?: string[]  // 约束条件
  successCriteria?: string[] // 成功标准
}

// 活动记录
interface Activity {
  type: 'tool_use' | 'thought' | 'result'
  timestamp: number
  summary: string
  details?: string
}
```

### 2.2 工人实现

```typescript
class BaseWorker implements Worker {
  readonly id: string
  readonly type: WorkerType
  status: WorkerStatus = 'idle'
  currentTask?: string
  
  private llm: LLMProvider
  private tools: Map<string, Tool>
  private messages: Message[] = []
  private activities: Activity[] = []
  private output: string = ''
  private abortController?: AbortController
  
  constructor(
    id: string,
    type: WorkerType,
    llm: LLMProvider,
    tools: Tool[]
  ) {
    this.id = id
    this.type = type
    this.llm = llm
    this.tools = new Map(tools.map(t => [t.name, t]))
  }
  
  // 分配任务
  async assign(spec: TaskSpec): Promise<void> {
    if (this.status !== 'idle') {
      throw new Error(`Worker ${this.id} is not idle`)
    }
    
    this.status = 'busy'
    this.currentTask = spec.subject
    this.abortController = new AbortController()
    
    // 构建初始消息
    this.messages = [
      {
        role: 'system',
        content: this.buildSystemPrompt(spec),
      },
      {
        role: 'user',
        content: this.buildUserPrompt(spec),
      },
    ]
    
    // 启动执行循环
    this.executeLoop().catch(err => {
      logger.error(`Worker ${this.id} execution failed:`, err)
      this.status = 'error'
    })
  }
  
  // 继续任务 (给正在执行的工人发送新指令)
  async continue(message: string): Promise<void> {
    if (this.status !== 'busy') {
      throw new Error(`Worker ${this.id} is not busy`)
    }
    
    // 添加用户消息
    this.messages.push({
      role: 'user',
      content: message,
    })
    
    // 继续执行循环
    this.executeLoop().catch(err => {
      logger.error(`Worker ${this.id} execution failed:`, err)
      this.status = 'error'
    })
  }
  
  // 停止工人
  async stop(): Promise<void> {
    this.abortController?.abort()
    this.status = 'idle'
    this.currentTask = undefined
  }
  
  // 执行循环
  private async executeLoop(): Promise<void> {
    const maxTurns = 20
    let turns = 0
    
    while (turns < maxTurns && this.status === 'busy') {
      turns++
      
      // 1. 调用 LLM
      const response = await this.llm.chat(this.messages, {
        tools: Array.from(this.tools.values()),
        signal: this.abortController?.signal,
      })
      
      // 2. 记录助手消息
      this.messages.push({
        role: 'assistant',
        content: response.content,
      })
      
      // 3. 处理工具调用
      if (response.toolCalls?.length) {
        for (const toolCall of response.toolCalls) {
          await this.executeToolCall(toolCall)
        }
      }
      
      // 4. 检查是否完成
      if (this.isTaskComplete(response.content)) {
        this.output = response.content
        this.status = 'idle'
        break
      }
    }
    
    if (turns >= maxTurns) {
      logger.warn(`Worker ${this.id} reached max turns`)
      this.output = 'Reached maximum execution turns'
      this.status = 'idle'
    }
  }
  
  // 执行工具调用
  private async executeToolCall(toolCall: ToolCall): Promise<void> {
    const tool = this.tools.get(toolCall.name)
    
    if (!tool) {
      this.messages.push({
        role: 'tool',
        toolCallId: toolCall.id,
        content: `Error: Tool '${toolCall.name}' not found`,
      })
      return
    }
    
    // 记录活动
    this.activities.push({
      type: 'tool_use',
      timestamp: Date.now(),
      summary: `Using ${toolCall.name}`,
      details: JSON.stringify(toolCall.arguments),
    })
    
    try {
      // 执行工具
      const result = await tool.execute(toolCall.arguments, {
        workerId: this.id,
        signal: this.abortController?.signal,
      })
      
      // 记录结果
      this.messages.push({
        role: 'tool',
        toolCallId: toolCall.id,
        content: result.content,
      })
      
      // 记录活动
      this.activities.push({
        type: 'result',
        timestamp: Date.now(),
        summary: `${toolCall.name} completed`,
        details: result.summary,
      })
    } catch (error) {
      // 记录错误
      this.messages.push({
        role: 'tool',
        toolCallId: toolCall.id,
        content: `Error: ${error instanceof Error ? error.message : 'Unknown error'}`,
      })
      
      this.activities.push({
        type: 'result',
        timestamp: Date.now(),
        summary: `${toolCall.name} failed`,
        details: error instanceof Error ? error.message : 'Unknown error',
      })
    }
  }
  
  // 构建系统提示词
  private buildSystemPrompt(spec: TaskSpec): string {
    return `You are a worker agent executing a task.

## Your Task
**Subject**: ${spec.subject}
**Description**: ${spec.description}

${spec.context ? `## Context
${spec.context}` : ''}

${spec.constraints?.length ? `## Constraints
${spec.constraints.join('\n')}` : ''}

${spec.successCriteria?.length ? `## Success Criteria
${spec.successCriteria.join('\n')}` : ''}

## Available Tools
You have access to the following tools:
${Array.from(this.tools.values()).map(t => `- ${t.name}: ${t.description}`).join('\n')}

## Output Format
When you complete the task, provide:
1. Summary of work done
2. List of modified files (if any)
3. Test results (if applicable)
4. Any issues encountered

## Important Rules
1. Do not ask for permission - just execute
2. Report errors immediately
3. Do not make assumptions - verify facts
4. Complete the task or explain why you cannot`
  }
  
  // 构建用户提示词
  private buildUserPrompt(spec: TaskSpec): string {
    return `Please execute the following task:

**Task**: ${spec.subject}

${spec.description}

Start now.`
  }
  
  // 检查任务是否完成
  private isTaskComplete(content: string): boolean {
    // 检查是否包含完成标记
    const completionPatterns = [
      /任务完成|task complete|done/i,
      /已经完成|completed|finished/i,
      /总结|summary/i,
    ]
    
    return completionPatterns.some(p => p.test(content))
  }
  
  // 获取状态
  getStatus(): WorkerStatus {
    return this.status
  }
  
  // 获取输出
  getOutput(): string {
    return this.output
  }
  
  // 获取活动
  getActivities(): Activity[] {
    return this.activities
  }
  
  // 清理
  async cleanup(): Promise<void> {
    await this.stop()
    this.messages = []
    this.activities = []
    this.output = ''
  }
}
```

## 3. 工人池管理

### 3.1 工人池实现

```typescript
class WorkerPool {
  private workers: Map<string, Worker> = new Set()
  private minWorkers: number
  private maxWorkers: number
  private idleTimeout: number
  
  constructor(config: WorkerPoolConfig) {
    this.minWorkers = config.minWorkers || 1
    this.maxWorkers = config.maxWorkers || 5
    this.idleTimeout = config.idleTimeout || 300000  // 5 分钟
  }
  
  // 获取工人 (有则复用，无则创建)
  async acquireWorker(type: WorkerType = 'general'): Promise<Worker> {
    // 1. 查找同类型的空闲工人
    for (const worker of this.workers.values()) {
      if (worker.status === 'idle' && 
          (worker.type === type || worker.type === 'general')) {
        return worker
      }
    }
    
    // 2. 如果可以创建新工人
    if (this.workers.size < this.maxWorkers) {
      const worker = await this.createWorker(type)
      this.workers.add(worker)
      return worker
    }
    
    // 3. 等待工人释放
    return this.waitForWorker()
  }
  
  // 创建工人
  private async createWorker(type: WorkerType): Promise<Worker> {
    const id = generateWorkerId(type)
    const llm = await this.createLLMProvider()
    const tools = await this.loadToolsForType(type)
    
    return new BaseWorker(id, type, llm, tools)
  }
  
  // 释放工人
  async releaseWorker(worker: Worker): Promise<void> {
    worker.status = 'idle'
    
    // 启动空闲超时计时器
    setTimeout(async () => {
      if (worker.status === 'idle' && 
          this.workers.size > this.minWorkers) {
        await worker.cleanup()
        this.workers.delete(worker)
        logger.info(`Cleaned up idle worker ${worker.id}`)
      }
    }, this.idleTimeout)
  }
  
  // 等待工人
  private async waitForWorker(): Promise<Worker> {
    return new Promise((resolve) => {
      const checkInterval = setInterval(() => {
        for (const worker of this.workers.values()) {
          if (worker.status === 'idle') {
            clearInterval(checkInterval)
            resolve(worker)
            return
          }
        }
      }, 1000)
    })
  }
  
  // 为类型加载工具
  private async loadToolsForType(type: WorkerType): Promise<Tool[]> {
    const toolNames = WORKER_TOOL_CONFIGS[type]
    
    const tools: Tool[] = []
    for (const name of toolNames) {
      const tool = await this.loadTool(name)
      if (tool) {
        tools.push(tool)
      }
    }
    
    return tools
  }
  
  // 加载单个工具
  private async loadTool(name: string): Promise<Tool | null> {
    try {
      // 动态加载工具
      const module = await import(`../tools/${name}Tool.js`)
      return module[name + 'Tool'] as Tool
    } catch (error) {
      logger.error(`Failed to load tool ${name}:`, error)
      return null
    }
  }
  
  // 列出工人
  listWorkers(): Worker[] {
    return Array.from(this.workers.values())
  }
  
  // 获取工人状态
  getStats(): WorkerPoolStats {
    const workers = this.listWorkers()
    
    return {
      total: workers.length,
      idle: workers.filter(w => w.status === 'idle').length,
      busy: workers.filter(w => w.status === 'busy').length,
      paused: workers.filter(w => w.status === 'paused').length,
      error: workers.filter(w => w.status === 'error').length,
    }
  }
  
  // 是否可以创建工人
  async canCreateWorker(): Promise<boolean> {
    return this.workers.size < this.maxWorkers
  }
  
  // 关闭所有工人
  async shutdown(): Promise<void> {
    logger.info(`Shutting down ${this.workers.size} workers`)
    
    for (const worker of this.workers.values()) {
      await worker.stop()
      await worker.cleanup()
    }
    
    this.workers.clear()
  }
}

interface WorkerPoolConfig {
  minWorkers?: number
  maxWorkers?: number
  idleTimeout?: number
}

interface WorkerPoolStats {
  total: number
  idle: number
  busy: number
  paused: number
  error: number
}
```

```typescript
class WorkerMetrics {
  // 记录工具调用
  recordToolUse(workerId: string, toolName: string, duration: number): void {
    metrics.histogram('worker.tool.duration', duration, {
      worker_id: workerId,
      tool_name: toolName,
    })
    
    metrics.counter('worker.tool.calls', 1, {
      worker_id: workerId,
      tool_name: toolName,
    })
  }
  
  // 记录 LLM 调用
  recordLLMCall(workerId: string, tokens: number, cost: number): void {
    metrics.histogram('worker.llm.tokens', tokens, {
      worker_id: workerId,
    })
    
    metrics.counter('worker.llm.cost', cost, {
      worker_id: workerId,
    })
  }
  
  // 记录任务完成
  recordTaskComplete(workerId: string, duration: number, turns: number): void {
    metrics.histogram('worker.task.duration', duration, {
      worker_id: workerId,
    })
    
    metrics.histogram('worker.task.turns', turns, {
      worker_id: workerId,
    })
    
    metrics.counter('worker.task.completed', 1, {
      worker_id: workerId,
    })
  }
  
  // 记录错误
  recordError(workerId: string, errorType: string): void {
    metrics.counter('worker.error', 1, {
      worker_id: workerId,
      error_type: errorType,
    })
  }
}
```

### 5.2 超时检测

```typescript
class WorkerTimeoutMonitor {
  private timeouts: Map<string, NodeJS.Timeout> = new Map()
  private readonly DEFAULT_TIMEOUT = 60 * 60 * 1000  // 60 分钟
  
  // 开始超时监控
  startMonitoring(workerId: string, timeoutMs?: number): void {
    const timeout = setTimeout(async () => {
      logger.warn(`Worker ${workerId} timed out`)
      
      // 获取工人当前活动
      const worker = await this.workerPool.getWorker(workerId)
      if (worker) {
        const activities = worker.getActivities()
        const lastActivity = activities[activities.length - 1]
        
        logger.warn(
          `Worker ${workerId} last activity: ${lastActivity?.summary}`
        )
      }
      
      // 停止工人
      await this.workerPool.killWorker(workerId)
      
      // 发布超时事件
      eventBus.publish('worker:timeout', { workerId })
    }, timeoutMs || this.DEFAULT_TIMEOUT)
    
    this.timeouts.set(workerId, timeout)
  }
  
  // 停止超时监控
  stopMonitoring(workerId: string): void {
    const timeout = this.timeouts.get(workerId)
    if (timeout) {
      clearTimeout(timeout)
      this.timeouts.delete(workerId)
    }
  }
  
  // 重置超时 (工人有活动时调用)
  resetTimeout(workerId: string): void {
    this.stopMonitoring(workerId)
    this.startMonitoring(workerId)
  }
}
```

## 6. 工人失败处理

### 6.1 失败类型

```typescript
enum WorkerFailureType {
  LLM_ERROR = 'llm_error',              // LLM API 错误
  TOOL_ERROR = 'tool_error',            // 工具执行错误
  TIMEOUT = 'timeout',                  // 超时
  MAX_TURNS = 'max_turns',              // 达到最大轮次
  ABORTED = 'aborted',                  // 被中止
  OUT_OF_TOKENS = 'out_of_tokens',      // Token 耗尽
}

interface WorkerFailure {
  workerId: string
  type: WorkerFailureType
  error?: string
  timestamp: number
  recoverable: boolean
  suggestedAction: 'retry' | 'continue' | 'abort'
}
```

### 6.2 失败恢复策略

```typescript
class WorkerRecovery {
  // 处理工人失败
  async handleFailure(failure: WorkerFailure): Promise<RecoveryAction> {
    switch (failure.type) {
      case WorkerFailureType.LLM_ERROR:
        return this.handleLLMError(failure)
        
      case WorkerFailureType.TOOL_ERROR:
        return this.handleToolError(failure)
        
      case WorkerFailureType.TIMEOUT:
        return this.handleTimeout(failure)
        
      case WorkerFailureType.MAX_TURNS:
        return this.handleMaxTurns(failure)
        
      case WorkerFailureType.ABORTED:
        return this.handleAborted(failure)
        
      case WorkerFailureType.OUT_OF_TOKENS:
        return this.handleOutOfTokens(failure)
    }
  }
  
  // 处理 LLM 错误
  private handleLLMError(failure: WorkerFailure): RecoveryAction {
    // API 错误通常是可恢复的
    return {
      action: 'retry',
      delay: 5000,  // 5 秒后重试
      reason: 'LLM API error, retrying',
    }
  }
  
  // 处理工具错误
  private handleToolError(failure: WorkerFailure): RecoveryAction {
    // 工具错误可能是不可恢复的
    const isRecoverable = this.isToolErrorRecoverable(failure.error)
    
    if (isRecoverable) {
      return {
        action: 'continue',
        message: 'The tool failed. Please try a different approach.',
        reason: 'Tool error, continuing with different approach',
      }
    }
    
    return {
      action: 'abort',
      reason: 'Unrecoverable tool error',
    }
  }
  
  // 处理超时
  private handleTimeout(failure: WorkerFailure): RecoveryAction {
    // 超时可能是任务太复杂
    return {
      action: 'retry',
      delay: 0,
      modifyPrompt: (prompt) => {
        // 添加更具体的指导
        return prompt + '\n\nPlease complete the task in fewer steps.'
      },
      reason: 'Task timed out, retrying with more guidance',
    }
  }
  
  // 处理最大轮次
  private handleMaxTurns(failure: WorkerFailure): RecoveryAction {
    // 达到最大轮次说明任务可能太复杂或陷入循环
    return {
      action: 'continue',
      message: 'You have reached the maximum number of turns. Please summarize what you have done and what remains.',
      reason: 'Max turns reached, requesting summary',
    }
  }
  
  // 处理被中止
  private handleAborted(failure: WorkerFailure): RecoveryAction {
    // 被用户中止，不需要恢复
    return {
      action: 'abort',
      reason: 'Task was aborted by user',
    }
  }
  
  // 处理 Token 耗尽
  private handleOutOfTokens(failure: WorkerFailure): RecoveryAction {
    // Token 耗尽可能需要压缩上下文
    return {
      action: 'continue',
      message: 'We are running low on tokens. Please proceed more efficiently.',
      reason: 'Out of tokens, requesting efficient approach',
    }
  }
  
  // 判断工具错误是否可恢复
  private isToolErrorRecoverable(error?: string): boolean {
    if (!error) return false
    
    const recoverablePatterns = [
      /file not found/i,
      /permission denied/i,
      /network error/i,
    ]
    
    return recoverablePatterns.some(p => p.test(error))
  }
}

interface RecoveryAction {
  action: 'retry' | 'continue' | 'abort'
  delay?: number
  message?: string
  modifyPrompt?: (prompt: string) => string
  reason: string
}
```

## 7. 工人生命周期管理

### 7.1 生命周期事件

```
创建 (Create)
    │
    ▼
空闲 (Idle) ──► 分配任务 (Assign)
    ▲               │
    │               ▼
    │          执行中 (Busy)
    │               │
    │               ├──► 完成任务 ──► 空闲
    │               │
    │               ├──► 失败 ──► 恢复/中止
    │               │
    │               └──► 被中止 ──► 空闲
    │
    ▼
清理 (Cleanup)
    │
    ▼
销毁 (Destroy)
```

### 7.2 生命周期管理

```typescript
class WorkerLifecycleManager {
  // 记录生命周期事件
  recordEvent(worker: Worker, event: LifecycleEvent): void {
    logger.info(`Worker ${worker.id}: ${event.type}`)
    
    // 发送到事件总线
    eventBus.publish('worker:lifecycle', {
      workerId: worker.id,
      event: event.type,
      timestamp: Date.now(),
      metadata: event.metadata,
    })
    
    // 持久化到日志
    this.appendLifecycleLog(worker.id, event)
  }
  
  // 追加生命周期日志
  private async appendLifecycleLog(
    workerId: string,
    event: LifecycleEvent
  ): Promise<void> {
    const logEntry = JSON.stringify({
      timestamp: Date.now(),
      workerId,
      event: event.type,
      metadata: event.metadata,
    })
    
    await appendFile('worker-lifecycle.jsonl', logEntry + '\n')
  }
  
  // 获取工人历史
  async getWorkerHistory(workerId: string): Promise<LifecycleEvent[]> {
    const logs = await read('worker-lifecycle.jsonl', 'utf-8')
    
    return logs
      .split('\n')
      .filter(line => line.trim())
      .map(line => JSON.parse(line))
      .filter(entry => entry.workerId === workerId)
      .map(entry => entry.event)
  }
  
  // 清理工人 (长时间空闲)
  async cleanupIdleWorkers(maxIdleTime: number): Promise<number> {
    const workers = this.workerPool.listWorkers()
    let cleaned = 0
    
    for (const worker of workers) {
      const idleTime = this.getIdleTime(worker)
      
      if (idleTime > maxIdleTime) {
        await worker.cleanup()
        this.workerPool.removeWorker(worker.id)
        cleaned++
        
        this.recordEvent(worker, {
          type: 'cleanup',
          metadata: { reason: 'idle_timeout', idleTime },
        })
      }
    }
    
    return cleaned
  }
  
  // 获取空闲时间
  private getIdleTime(worker: Worker): number {
    const activities = worker.getActivities()
    const lastActivity = activities[activities.length - 1]
    
    if (!lastActivity) {
      return Date.now() - worker.createdAt
    }
    
    return Date.now() - lastActivity.timestamp
  }
}

interface LifecycleEvent {
  type: 'create' | 'assign' | 'complete' | 'fail' | 'abort' | 'cleanup' | 'destroy'
  metadata?: Record<string, unknown>
}
```

## 9. 最佳实践

### 9.1 工人提示词

1. **自包含**：工人看不到对话历史，提示词必须包含所有必要信息
2. **具体路径**：包含具体的文件路径和行号
3. **明确标准**：说明什么是"完成"
4. **目的说明**：让工人理解任务的目的

### 9.2 工人管理

1. **优先复用**：有空闲工人优先复用，避免不必要的新建
2. **及时清理**：长时间空闲的工人及时清理
3. **类型匹配**：根据任务类型匹配合适的工人
4. **优雅关闭**：关闭时等待当前任务完成

### 9.3 失败处理

1. **区分类型**：区分可恢复和不可恢复的错误
2. **自动重试**：可恢复的错误自动重试
3. **请求帮助**：不可恢复的错误请求人工干预
4. **记录日志**：所有失败都要记录便于分析

```typescript
// 记忆提取
async function extractMemory(conversation: Conversation): Promise<Memory[]> {
  // 1. 分析对话内容
  const analysis = await analyze(conversation)
  
  // 2. 识别候选记忆
  const candidates = identifyCandidates(analysis)
  
  // 3. 过滤 (排除可从代码推导的信息)
  const filtered = candidates.filter(m => !isDerivable(m))
  
  // 4. 格式化
  return filtered.map(formatAsMemory)
}

// 记忆存储
async function saveMemory(memory: Memory): Promise<void> {
  // 1. 写入记忆文件
  const fileName = `${memory.type}_${snakeCase(memory.name)}.md`
  const content = formatMemoryFile(memory)
  await write(join(MEMORY_DIR, fileName), content)
  
  // 2. 更新索引
  await updateIndex(fileName, memory)
}

// 记忆检索
async function retrieveMemory(query: string): Promise<Memory[]> {
  // 1. 加载索引
  const index = await loadIndex()
  
  // 2. 语义检索 (或关键词检索)
  const relevant = index.filter(entry => 
    matches(entry, query)
  )
  
  // 3. 加载全文
  return Promise.all(relevant.map(loadMemory))
}
```

### 4.3 任务依赖管理

```typescript
// 任务依赖图
const taskGraph = {
  'task-1': { id: 'task-1', blockedBy: [], blocks: ['task-2'] },
  'task-2': { id: 'task-2', blockedBy: ['task-1'], blocks: ['task-3', 'task-4'] },
  'task-3': { id: 'task-3', blockedBy: ['task-2'], blocks: [] },
  'task-4': { id: 'task-4', blockedBy: ['task-2'], blocks: [] },
}

// 检查任务是否可以开始
function canStart(taskId: string, tasks: Map<string, Task>): boolean {
  const task = tasks.get(taskId)
  if (!task) return false
  if (task.status !== 'pending') return false
  
  // 检查所有依赖是否完成
  for (const depId of task.blockedBy) {
    const dep = tasks.get(depId)
    if (!dep || dep.status !== 'completed') {
      return false
    }
  }
  return true
}

// 检测循环依赖
function hasCircularDependency(taskId: string, tasks: Map<string, Task>): boolean {
  const visited = new Set<string>()
  const stack = new Set<string>()
  
  function dfs(id: string): boolean {
    if (stack.has(id)) return true  // 检测到循环
    if (visited.has(id)) return false
    
    visited.add(id)
    stack.add(id)
    
    const task = tasks.get(id)
    for (const depId of task?.blockedBy || []) {
      if (dfs(depId)) return true
    }
    
    stack.delete(id)
    return false
  }
  
  return dfs(taskId)
}
```

### 5.3 工人管理策略

```typescript
class WorkerPool {
  private workers: Map<string, Worker>
  private maxWorkers: number
  private minWorkers: number
  
  // 获取或创建工人
  async acquireWorker(): Promise<Worker> {
    // 1. 查找空闲工人
    for (const worker of this.workers.values()) {
      if (worker.status === 'idle') {
        return worker
      }
    }
    
    // 2. 如果没有空闲且未达上限，创建新工人
    if (this.workers.size < this.maxWorkers) {
      const worker = await this.createWorker()
      this.workers.set(worker.id, worker)
      return worker
    }
    
    // 3. 等待工人释放
    return this.waitForWorker()
  }
  
  // 释放工人
  async releaseWorker(workerId: string): Promise<void> {
    const worker = this.workers.get(workerId)
    if (worker) {
      worker.currentTask = undefined
      worker.status = 'idle'
    }
  }
  
  // 清理空闲工人 (保留最小数量)
  async cleanupIdleWorkers(): Promise<void> {
    const idleWorkers = Array.from(this.workers.values())
      .filter(w => w.status === 'idle')
    
    const toRemove = idleWorkers.slice(
      this.minWorkers,
      idleWorkers.length
    )
    
    for (const worker of toRemove) {
      await worker.cleanup()
      this.workers.delete(worker.id)
    }
  }
}
```

### 6.2 状态同步

```typescript
// 状态同步机制
class StateSync {
  // 发布 - 订阅模式
  private subscribers: Map<string, Set<Handler>>
  
  // 订阅状态变化
  subscribe(entityType: 'task' | 'worker', handler: Handler): void {
    if (!this.subscribers.has(entityType)) {
      this.subscribers.set(entityType, new Set())
    }
    this.subscribers.get(entityType)!.add(handler)
  }
  
  // 发布状态变化
  publish(entityType: 'task' | 'worker', entityId: string, newState: any): void {
    const handlers = this.subscribers.get(entityType)
    handlers?.forEach(h => h(entityId, newState))
  }
}

// 使用示例
const sync = new StateSync()

// 管家订阅任务状态变化
sync.subscribe('task', (taskId, newStatus) => {
  if (newStatus === 'completed') {
    butler.verifyResult(taskId)
  }
})

// 任务模块发布状态变化
task.updateStatus('completed')
sync.publish('task', task.id, 'completed')
```

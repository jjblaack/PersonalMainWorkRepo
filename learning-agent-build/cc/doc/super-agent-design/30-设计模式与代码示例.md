## 1. 工具接口模式

### 1.1 模式概述

**用途**：定义统一的工具接口，使所有工具具有一致的调用方式、权限检查和结果返回。

**实现位置**：`src/Tool.ts`, `src/tools/*/`

### 1.2 核心接口

```typescript
// 工具接口定义
interface Tool {
  name: string
  description: string
  inputSchema: ToolInputJSONSchema
  
  // 核心执行方法
  execute(
    input: unknown,
    context: ToolUseContext
  ): Promise<ToolResult>
  
  // 可选方法
  validateInput?(input: unknown, context: ToolUseContext): Promise<ValidationResult>
  checkPermissions?(input: unknown, context: ToolUseContext): Promise<PermissionResult>

  success: boolean
  content: ContentBlock[]
  error?: string
  data?: unknown
}
```

### 1.3 工具实现模板

```typescript
// 工具实现示例 (基于 FileReadTool)
export const FileReadTool = buildTool({
  name: FILE_READ_TOOL_NAME,
  searchHint: 'read files, images, PDFs',
  strict: true,
  
  // 输入 Schema
  get inputSchema(): InputSchema {
    return z.object({
      file_path: z.string().describe('Path to the file to read'),
      pages: z.array(z.number()).optional().describe('Pages to read (for PDFs)'),
    })
  },
  
  // 输出 Schema
  get outputSchema(): OutputSchema {
    return z.object({
      content: z.string(),
      metadata: z.object({
        size: z.number(),
        modified: z.string(),
      }).optional(),
    })
  },
  
  // 工具描述
  async description() {
    return 'Read the contents of a file at the specified path'
  },
  
  // 输入验证
  async validateInput(input, context) {
    const { file_path } = input
    
    // 检查文件是否存在
    const exists = await fs.exists(file_path)
    if (!exists) {
      return { result: false, error: 'File does not exist' }
    }
    
    // 检查文件类型
    const isAllowed = await isFileTypeAllowed(file_path)
    if (!isAllowed) {
      return { result: false, error: 'File type not allowed' }
    }
    
    return { result: true }
  },
  
  // 核心执行逻辑
  async call(input, context) {
    const { file_path, pages } = input
    
    // 读取文件
    const content = await fs.readFile(file_path, 'utf-8')
    
    // 处理分页 (如果是 PDF)
    const processedContent = pages 
      ? await extractPages(content, pages)
      : content
    
    return {
      success: true,
      content: [{
        type: 'text',
        text: processedContent,
      }],
      data: { content: processedContent },
    }
  },
} satisfies ToolDef<InputSchema, Output>)
```

### 1.4 buildTool 工厂函数

```typescript
// 工具工厂函数 (简化版)
function buildTool<T, U>(def: {
  name: string
  searchHint?: string
  strict?: boolean
  inputSchema: z.ZodType<T>
  outputSchema?: z.ZodType<U>
  description: () => Promise<string>
  validateInput?: (input: T, context: ToolUseContext) => Promise<ValidationResult>
  call: (input: T, context: ToolUseContext) => Promise<ToolResult>
}): Tool {
  return {
    name: def.name,
    description: def.description(),
    inputSchema: def.inputSchema,
    
    async execute(input, context) {
      // 1. Schema 验证
      const validated = def.inputSchema.safeParse(input)
      if (!validated.success) {
        return {
          success: false,
          content: [],
          error: `Invalid input: ${validated.error.message}`,
        }
      }
      
      // 2. 自定义验证 (如果有)
      if (def.validateInput) {
        const validation = await def.validateInput(validated.data, context)
        if (!validation.result) {
          return {
            success: false,
            content: [],
            error: validation.error,
          }
        }
      }
      
      // 3. 执行核心逻辑
      return def.call(validated.data, context)
    },
  }
}
```

### 3.3 状态存储

```typescript
// 状态存储类
class AppStateStore {
  private state: AppState
  private listeners: Set<() => void> = new Set()
  
  constructor(initialState: AppState) {
    this.state = initialState
  }
  
  // 获取状态
  getState(): AppState {
    return this.state
  }
  
  // 更新状态 (使用函数以确保不可变性)
  update(updater: (prev: AppState) => AppState): void {
    const prevState = this.state
    this.state = updater(prevState)
    this.notifyListeners()
  }
  
  // 订阅状态变化
  subscribe(listener: () => void): () => void {
    this.listeners.add(listener)
    return () => this.listeners.delete(listener)
  }
  
  // 通知监听器
  private notifyListeners(): void {
    for (const listener of this.listeners) {
      listener()
    }
  }
}
```

### 4.3 错误分类工具

```typescript
// 错误分类函数
function categorizeError(error: unknown): ErrorCategory {
  if (error instanceof AbortError) {
    return { type: 'abort', recoverable: false }
  }
  
  if (error instanceof TimeoutError) {
    return { type: 'timeout', recoverable: true }
  }
  
  if (error instanceof RetryableAPIError) {
    return { 
      type: 'retryable', 
      recoverable: true,
      retryAfter: error.retryAfter,
    }
  }
  
  if (error instanceof ResourceExhaustedError) {
    return { type: 'resource', recoverable: false }
  }
  
  // 默认：未知错误，尝试重试
  return { type: 'unknown', recoverable: true }
}

// 错误处理装饰器
function withRetry<T extends (...args: any[]) => Promise<any>>(
  fn: T,
  maxRetries: number = 3,
  backoffMs: number[] = [1000, 2000, 5000],
): T {
  return (async (...args: any[]) => {
    let lastError: Error
    
    for (let i = 0; i < maxRetries; i++) {
      try {
        return await fn(...args)
      } catch (error) {
        lastError = error as Error
        const category = categorizeError(error)
        
        if (!category.recoverable) {
          throw error
        }
        
        // 等待后重试
        const delay = backoffMs[Math.min(i, backoffMs.length - 1)]
        await sleep(delay)
      }
    }
    
    throw lastError!
  }) as T
}
```

## 5. 异步并发模式

### 5.1 模式概述

**用途**：管理异步操作、并发控制和取消机制。

### 5.2 Stream 模式

```typescript
// 异步数据流
class Stream<T> implements AsyncIterator<T> {
  private queue: T[] = []
  private resolve?: (value: IteratorResult<T>) => void
  private reject?: (error: unknown) => void
  private done = false
  private error?: unknown
  
  // 消费端：获取下一个值
  async next(): Promise<IteratorResult<T>> {
    if (this.queue.length > 0) {
      return { done: false, value: this.queue.shift()! }
    }
    
    if (this.done) {
      return { done: true, value: undefined }
    }
    
    if (this.error) {
      throw this.error
    }
    
    // 等待数据
    return new Promise((resolve, reject) => {
      this.resolve = resolve
      this.reject = reject
    })
  }
  
  // 生产端：推送数据
  enqueue(value: T): void {
    if (this.resolve) {
      const resolve = this.resolve
      this.resolve = undefined
      this.reject = undefined
      resolve({ done: false, value })
    } else {
      this.queue.push(value)
    }
  }
  
  // 生产端：结束流
  finish(): void {
    this.done = true
    if (this.resolve) {
      this.resolve({ done: true, value: undefined })
    }
  }
  
  // 生产端：报告错误
  fail(error: unknown): void {
    this.error = error
    if (this.reject) {
      this.reject(error)
    }
  }
}
```

### 5.3 并发控制

```typescript
// 并发限制器
class ConcurrencyLimiter {
  private running = 0
  private queue: Array<() => void> = []
  
  constructor(private maxConcurrency: number) {}
  
  async run<T>(fn: () => Promise<T>): Promise<T> {
    // 等待可用槽位
    if (this.running >= this.maxConcurrency) {
      await new Promise<void>(resolve => {
        this.queue.push(resolve)
      })
    }
    
    this.running++
    
    try {
      return await fn()
    } finally {
      this.running--
      
      // 唤醒下一个等待的任务
      const next = this.queue.shift()
      if (next) {
        next()
      }
    }
  }
}

// 使用示例
const limiter = new ConcurrencyLimiter(3)

// 并行执行，但最多 3 个并发
const results = await Promise.all(
  tasks.map(task => limiter.run(() => executeTask(task)))
)
```

### 5.4 取消机制

```typescript
// 取消 Token
class CancellationToken {
  private cancelled = false
  private listeners: Set<() => void> = new Set()
  
  get isCancelled(): boolean {
    return this.cancelled
  }
  
  cancel(): void {
    if (this.cancelled) return
    this.cancelled = true
    
    for (const listener of this.listeners) {
      listener()
    }
  }
  
  throwIfCancelled(): void {
    if (this.cancelled) {
      throw new Error('Operation cancelled')
    }
  }
  
  register(callback: () => void): () => void {
    this.listeners.add(callback)
    return () => this.listeners.delete(callback)
  }
}

// 使用示例
async function longRunningOperation(token: CancellationToken): Promise<void> {
  for (let i = 0; i < 100; i++) {
    token.throwIfCancelled()
    await doWork()
  }
}

// 创建带取消的操作
const token = new CancellationToken()

const promise = longRunningOperation(token)

// 需要取消时
token.cancel()
```

## 7. 事件总线模式

### 7.1 模式概述

**用途**：模块间解耦通信，支持发布 - 订阅模式。

### 7.2 事件总线实现

```typescript
// 事件定义
type EventType = 
  | 'task:created'
  | 'task:started'
  | 'task:completed'
  | 'task:failed'
  | 'worker:spawned'
  | 'worker:stopped'
  | 'memory:saved'

interface Event {
  type: EventType
  payload: unknown
  timestamp: number
}

// 事件总线
class EventBus {
  private listeners: Map<EventType, Set<(event: Event) => void>> = new Map()
  
  // 发布事件
  publish(type: EventType, payload: unknown): void {
    const event: Event = {
      type,
      payload,
      timestamp: Date.now(),
    }
    
    const listeners = this.listeners.get(type)
    listeners?.forEach(fn => fn(event))
    
    // 记录事件 (用于调试)
    this.logEvent(event)
  }
  
  // 订阅事件
  subscribe(type: EventType, handler: (event: Event) => void): () => void {
    if (!this.listeners.has(type)) {
      this.listeners.set(type, new Set())
    }
    
    this.listeners.get(type)!.add(handler)
    
    // 返回取消订阅函数
    return () => {
      this.listeners.get(type)?.delete(handler)
    }
  }
  
  // 清空事件
  clear(): void {
    this.listeners.clear()
  }
  
  private logEvent(event: Event): void {
    // 可选：记录到文件或发送到遥测
  }
}

// 使用示例
const eventBus = new EventBus()

// 订阅任务完成事件
eventBus.subscribe('task:completed', (event) => {
  const { taskId, output } = event.payload as { taskId: string, output: string }
  logger.info(`Task ${taskId} completed with output: ${output}`)
})

// 发布事件
eventBus.publish('task:created', { taskId: 'task-123', subject: 'Fix bug' })
```

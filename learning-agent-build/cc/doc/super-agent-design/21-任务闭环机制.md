> 从任务创建、执行、验收到归档的完整生命周期管理，确保每个任务有始有终。

## 1. 任务状态机

### 1.1 状态定义

```typescript
type TaskStatus =
  | 'pending'       // 等待执行
  | 'in_progress'   // 正在执行
  | 'completed'     // 已完成
  | 'failed'        // 执行失败
  | 'killed'        // 被停止
  | 'deleted'       // 已删除

// 终端状态（不可逆）
const TERMINAL_STATUSES = ['completed', 'failed', 'killed', 'deleted']

function isTerminalStatus(status: TaskStatus): boolean {
  return TERMINAL_STATUSES.includes(status)
}
```

### 1.3 状态转换规则

```typescript
const VALID_TRANSITIONS: Record<TaskStatus, TaskStatus[]> = {
  'pending': ['in_progress', 'deleted'],
  'in_progress': ['completed', 'failed', 'killed', 'pending'],
  'completed': ['pending', 'deleted'],  // 可重新打开
  'failed': ['pending', 'deleted'],     // 可重试
  'killed': ['pending', 'deleted'],     // 可重新启动
  'deleted': [],  // 终端状态，不可转换
}

function validateTransition(from: TaskStatus, to: TaskStatus): boolean {
  return VALID_TRANSITIONS[from]?.includes(to) ?? false
}

function updateTaskStatus(task: Task, newStatus: TaskStatus): Task {
  if (!validateTransition(task.status, newStatus)) {
    throw new Error(
      `Invalid transition from ${task.status} to ${newStatus}`
    )
  }
  
  return {
    ...task,
    status: newStatus,
    updatedAt: Date.now(),
    endTime: isTerminalStatus(newStatus) ? Date.now() : task.endTime,
  }
}
```

```typescript
interface Task {
  // === 基本信息 ===
  id: string              // 任务 ID (唯一标识)
```

```typescript
function generateTaskId(type: TaskType): string {
  const prefix = TASK_ID_PREFIXES[type] ?? 'x'
  const bytes = randomBytes(8)
  let id = prefix
  for (let i = 0; i < 8; i++) {
    id += TASK_ID_ALPHABET[bytes[i]! % TASK_ID_ALPHABET.length]
  }
  return id
}
```

```typescript
// 检查任务是否可以开始
async function canStartTask(taskId: string): Promise<boolean> {
  const task = await getTask(taskId)
  if (!task) return false
  if (task.status !== 'pending') return false
  
  // 检查所有依赖是否完成
  for (const depId of task.blockedBy) {
    const dep = await getTask(depId)
    if (!dep || dep.status !== 'completed') {
      return false
    }
  }
  
  return true
}

// 获取可执行的任务（所有依赖已完成的待执行任务）
async function getExecutableTasks(): Promise<Task[]> {
  const pendingTasks = await listTasks({ status: 'pending' })
  const executable: Task[] = []
  
  for (const task of pendingTasks) {
    if (await canStartTask(task.id)) {
      executable.push(task)
    }
  }
  
  return executable
}

// 检测循环依赖
function hasCircularDependency(
  taskId: string,
  tasks: Map<string, Task>,
  visited = new Set<string>(),
  stack = new Set<string>()
): boolean {
  if (stack.has(taskId)) return true  // 检测到循环
  if (visited.has(taskId)) return false
  
  visited.add(taskId)
  stack.add(taskId)
  
  const task = tasks.get(taskId)
  if (!task) return false
  
  for (const depId of task.blockedBy) {
    if (hasCircularDependency(depId, tasks, visited, stack)) {
      return true
    }
  }
  
  stack.delete(taskId)
  return false
}
```

### 4.2 任务执行代码示例

```typescript
class TaskManager {
  private tasks: Map<string, Task>
  private workerPool: WorkerPool
  private eventEmitter: EventEmitter
  
  // 创建任务
  async createTask(spec: TaskSpec): Promise<Task> {
    const task: Task = {
      id: generateTaskId(spec.type),
      subject: spec.subject,
      description: spec.description,
      status: 'pending',
      blockedBy: spec.blockedBy ?? [],
      blocks: spec.blocks ?? [],
      priority: spec.priority ?? 'normal',
      createdAt: Date.now(),
      updatedAt: Date.now(),
      notified: false,
    }
    
    // 持久化
    await this.persistTask(task)
    this.tasks.set(task.id, task)
    
    // 触发事件
    this.eventEmitter.emit('task:created', task)
    
    return task
  }
  
  // 启动任务
  async startTask(taskId: string): Promise<void> {
    const task = this.tasks.get(taskId)
    if (!task) throw new Error(`Task ${taskId} not found`)
    
    // 检查依赖
    if (!await canStartTask(taskId)) {
      throw new Error(`Task ${taskId} has unmet dependencies`)
    }
    
    // 获取工人
    const worker = await this.workerPool.acquireWorker()
    
    // 更新状态
    task.status = 'in_progress'
    task.owner = worker.id
    task.startedAt = Date.now()
    task.updatedAt = Date.now()
    
    await this.persistTask(task)
    this.eventEmitter.emit('task:started', task)
    
    // 分配任务给工人
    worker.assign({
      subject: task.subject,
      description: task.description,
      context: await this.buildTaskContext(task),
    }).catch(err => {
      this.handleTaskError(taskId, err)
    })
  }
  
  // 完成任务
  async completeTask(taskId: string, output: string): Promise<void> {
    const task = this.tasks.get(taskId)
    if (!task) throw new Error(`Task ${taskId} not found`)
    
    task.status = 'completed'
    task.output = output
    task.endTime = Date.now()
    task.updatedAt = Date.now()
    task.notified = false
    
    await this.persistTask(task)
    this.eventEmitter.emit('task:completed', task)
    
    // 释放工人
    if (task.owner) {
      await this.workerPool.releaseWorker(task.owner)
    }
    
    // 通知依赖任务可以启动
    await this.notifyBlockedTasks(taskId)
  }
  
  // 处理失败
  async failTask(taskId: string, error: string): Promise<void> {
    const task = this.tasks.get(taskId)
    if (!task) throw new Error(`Task ${taskId} not found`)
    
    task.status = 'failed'
    task.metadata = {
      ...task.metadata,
      errorMessage: error,
    }
    task.endTime = Date.now()
    task.updatedAt = Date.now()
    
    await this.persistTask(task)
    this.eventEmitter.emit('task:failed', task)
    
    // 释放工人
    if (task.owner) {
      await this.workerPool.releaseWorker(task.owner)
    }
  }
  
  // 重试任务
  async retryTask(taskId: string): Promise<void> {
    const task = this.tasks.get(taskId)
    if (!task) throw new Error(`Task ${taskId} not found`)
    if (task.status !== 'failed') {
      throw new Error(`Can only retry failed tasks, got ${task.status}`)
    }
    
    task.status = 'pending'
    task.owner = undefined
    task.updatedAt = Date.now()
    task.metadata = {
      ...task.metadata,
      retryCount: (task.metadata?.retryCount ?? 0) + 1,
    }
    
    await this.persistTask(task)
    this.eventEmitter.emit('task:retried', task)
  }
}
```

### 5.2 验收流程

```typescript
class TaskVerifier {
  private checks: VerificationCheck[]
  
  // 添加检查
  addCheck(name: string, check: () => Promise<boolean>): void {
    this.checks.push({ name, passed: false })
  }
  
  // 执行验收
  async verify(task: Task): Promise<VerificationResult> {
    const results: VerificationCheck[] = []
    
    // 1. 检查输出存在
    const outputExists = await this.checkOutputExists(task)
    results.push({
      name: '输出存在',
      passed: outputExists,
    })
    
    // 2. 检查输出格式
    if (outputExists) {
      const outputValid = await this.checkOutputFormat(task)
      results.push({
        name: '输出格式正确',
        passed: outputValid,
      })
    }
    
    // 3. 运行测试 (如果任务包含测试)
    if (task.subject.includes('测试') || task.subject.includes('test')) {
      const testsPassed = await this.runTests(task)
      results.push({
        name: '测试通过',
        passed: testsPassed,
        details: testsPassed ? undefined : '部分测试失败',
      })
    }
    
    // 4. 类型检查 (如果是代码任务)
    if (task.subject.includes('实现') || task.subject.includes('代码')) {
      const typecheckPassed = await this.runTypecheck(task)
      results.push({
        name: '类型检查通过',
        passed: typecheckPassed,
        details: typecheckPassed ? undefined : '存在类型错误',
      })
    }
    
    // 汇总结果
    const passed = results.every(r => r.passed)
    
    return {
      passed,
      checks: results,
      summary: passed 
        ? '所有验收检查通过' 
        : `${results.filter(r => !r.passed).length} 项检查失败`,
    }
  }
  
  private async checkOutputExists(task: Task): Promise<boolean> {
    if (!task.outputFile) return false
    try {
      await access(task.outputFile)
      return true
    } catch {
      return false
    }
  }
  
  private async checkOutputFormat(task: Task): Promise<boolean> {
    if (!task.output) return false
    // 检查输出是否包含有意义的內容
    return task.output.trim().length > 0
  }

    // 这可能需要解析任务输出找到测试文件

  }
  
  private async runTypecheck(task: Task): Promise<boolean> {
    // 运行类型检查
    return true  // 简化
  }
}
```

## 6. 任务失败处理

### 6.1 失败类型

```typescript
enum FailureType {
  EXECUTION_ERROR = 'execution_error',     // 执行错误
  TIMEOUT = 'timeout',                      // 超时
  RESOURCE_EXHAUSTED = 'resource_exhausted', // 资源耗尽

}

interface TaskFailure {
  taskId: string
  type: FailureType
  error: string
  timestamp: number
  recoverable: boolean
  suggestedAction: 'retry' | 'fix_and_retry' | 'abort'
}
```

### 6.2 失败恢复策略

```typescript
class TaskRecovery {
  // 处理失败
  async handleFailure(failure: TaskFailure): Promise<void> {
    const task = await getTask(failure.taskId)
    if (!task) return
    
    switch (failure.type) {
      case FailureType.EXECUTION_ERROR:
        await this.handleExecutionError(task, failure)
        break
        
      case FailureType.TIMEOUT:
        await this.handleTimeout(task, failure)
        break
        
      case FailureType.RESOURCE_EXHAUSTED:
        await this.handleResourceExhausted(task, failure)
        break
        
      case FailureType.DEPENDENCY_FAILED:
        await this.handleDependencyFailed(task, failure)
        break
        
      case FailureType.VERIFICATION_FAILED:
        await this.handleVerificationFailed(task, failure)
        break
        
      case FailureType.CANCELLED:
        await this.handleCancelled(task, failure)
        break
    }
  }
  
  // 执行错误处理
  private async handleExecutionError(
    task: Task,
    failure: TaskFailure
  ): Promise<void> {
    const retryCount = task.metadata?.retryCount ?? 0
    
    if (retryCount < MAX_RETRIES) {
      // 可恢复，自动重试
      await taskManager.retryTask(task.id)
    } else {
      // 超过最大重试次数，需要人工干预
      await taskManager.failTask(task.id, failure.error)
      await this.notifyUser(task, failure)
    }
  }
  
  // 超时处理
  private async handleTimeout(
    task: Task,
    failure: TaskFailure
  ): Promise<void> {
    // 停止工人
    if (task.owner) {
      await workerPool.killWorker(task.owner)
    }
    
    // 判断是否可以重试
    if (failure.recoverable) {
      await taskManager.retryTask(task.id)
    } else {
      await taskManager.failTask(task.id, 'Task timed out')
    }
  }
  
  // 依赖失败处理
  private async handleDependencyFailed(
    task: Task,
    failure: TaskFailure
  ): Promise<void> {
    // 依赖失败通常不可恢复
    await taskManager.failTask(
      task.id,
      `Dependency failed: ${failure.error}`
    )
    await this.notifyUser(task, failure)
  }
  
  // 验收失败处理
  private async handleVerificationFailed(
    task: Task,
    failure: TaskFailure
  ): Promise<void> {
    // 验收失败可能是代码问题，需要修复
    // 继续同一工人进行修复
    if (task.owner) {
      await workerPool.continueWorker(
        task.owner,
        `验收失败：${failure.error}。请修复问题。`
      )
    } else {
      await taskManager.retryTask(task.id)
    }
  }
}
```

### 7.2 输出持久化

```typescript
// 写入输出
async function writeTaskOutput(
  taskId: string,
  content: string

  const outputPath = getTaskOutputPath(taskId)
  const formatted = formatTaskOutput(
    await getTask(taskId),

// 流式追加输出 (用于长任务)
async function appendTaskOutput(
  taskId: string,
  chunk: string,
  offset: number
): Promise<void> {
  const outputPath = getTaskOutputPath(taskId)
  
  // 追加到文件
  await appendFile(outputPath, chunk)
  
  // 更新偏移
  const task = await getTask(taskId)
  task.outputOffset = offset + chunk.length
  await updateTask(task)
}

// 读取输出
async function readTaskOutput(taskId: string): Promise<string> {
  const outputPath = getTaskOutputPath(taskId)
  
  try {
    return await read(outputPath, 'utf-8')
  } catch (error) {
    throw new Error(`Cannot read output for task ${taskId}`)
  }
}
```

### 8.1 任务监控

```typescript
class TaskMonitor {
  private runningTasks: Map<string, TaskMonitorEntry>
  private checkInterval: number
  
  constructor(checkIntervalMs = 30000) {
    this.checkInterval = checkIntervalMs
    this.startMonitoring()
  }
  
  private startMonitoring(): void {
    setInterval(() => this.checkAllTasks(), this.checkInterval)
  }
  
  private async checkAllTasks(): Promise<void> {
    const runningTasks = await listTasks({
      status: 'in_progress',
    })
    
    for (const task of runningTasks) {
      await this.checkTask(task)
    }
  }
  
  private async checkTask(task: Task): Promise<void> {
    const elapsed = Date.now() - task.startedAt!
    const timeout = task.metadata?.estimatedDuration ?? DEFAULT_TIMEOUT
    
    if (elapsed > timeout) {
      // 超时警告
      await this.sendTimeoutWarning(task, elapsed)
    }
    
    // 检查工人是否存活
    if (task.owner) {
      const workerAlive = await workerPool.isWorkerAlive(task.owner)
      if (!workerAlive) {
        await this.handleWorkerDeath(task)
      }
    }
  }
}
```

## 9. 最佳实践

### 9.1 任务设计

1. **原子性**：每个任务应该是原子的，专注于单一目标
2. **明确完成标准**：任务描述中明确什么算"完成"
3. **合理估算时间**：给任务设置合理的预期时间
4. **依赖最小化**：减少任务间依赖，提高并行度

### 9.2 任务执行

1. **及时状态更新**：状态变化立即持久化
2. **输出流式记录**：长任务流式记录输出，避免丢失
3. **超时保护**：所有任务设置超时
4. **优雅失败**：失败时保留尽可能多的上下文

### 9.3 任务监督

1. **主动监控**：定期检查任务状态，不要被动等待
2. **早期预警**：超时就预警，不要等到完全超时
3. **自动恢复**：可恢复的错误自动重试
4. **人工介入**：不可恢复的错误及时通知人工

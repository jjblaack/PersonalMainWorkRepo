## 1. 24 小时运行的挑战

### 1.1 主要挑战

| 挑战 | 说明 | 影响 |
|------|------|------|
| 进程保活 | 防止崩溃、自动重启 | 进程意外终止 |
| 连接管理 | 断线重连、心跳保持 | 与服务器断开连接 |
| 状态恢复 | 崩溃后从断点继续 | 丢失工作进度 |
| 资源管理 | 内存泄漏防护、Token 预算 | 资源耗尽导致崩溃 |
| 日志管理 | 日志轮转、磁盘空间 | 磁盘写满 |
| 安全更新 | 不中断服务的更新 | 需要停机更新 |

### 2.2 主循环架构

```typescript
// 主循环
class SuperAgentDaemon {
  private running = false
  private reconnectAttempts = 0
  private readonly MAX_RECONNECT_ATTEMPTS = 10
  private readonly RECONNECT_BACKOFF_MS = [1000, 2000, 5000, 10000, 30000]
  
  // 启动守护进程
  async start(): Promise<void> {
    this.running = true
    
    // 注册信号处理
    this.setupSignalHandlers()
    
    // 主循环
    while (this.running) {
      try {
        await this.runLoop()
      } catch (error) {
        await this.handleError(error)
      }
    }
  }
  
  // 主循环
  private async runLoop(): Promise<void> {
    // 1. 初始化
    await this.initialize()
    
    // 2. 连接到服务器
    await this.connect()
    
    // 3. 重置重连计数
    this.reconnectAttempts = 0
    
    // 4. 进入工作循环
    while (this.running) {
      // 轮询新指令
      const instructions = await this.pollInstructions()
      
      for (const instruction of instructions) {
        await this.handleInstruction(instruction)
      }
      
      // 发送心跳
      await this.sendHeartbeat()
      
      // 检查资源
      await this.checkResources()
      
      // 短暂休眠
      await sleep(POLL_INTERVAL_MS)
    }
  }
  
  // 错误处理
  private async handleError(error: Error): Promise<void> {
    logger.error('Main loop error:', error)
    
    // 计算退避时间
    const backoff = this.getBackoffMs()
    
    logger.info(`Reconnecting in ${backoff}ms...`)
    await sleep(backoff)
    
    this.reconnectAttempts++
    
    if (this.reconnectAttempts >= this.MAX_RECONNECT_ATTEMPTS) {
      logger.error('Max reconnect attempts reached, exiting')
      this.running = false
    }
  }
  
  // 指数退避
  private getBackoffMs(): number {
    const index = Math.min(
      this.reconnectAttempts,
      this.RECONNECT_BACKOFF_MS.length - 1
    )
    return this.RECONNECT_BACKOFF_MS[index]
  }
  
  // 信号处理
  private setupSignalHandlers(): void {
    process.on('SIGTERM', () => {
      logger.info('Received SIGTERM, shutting down gracefully')
      this.running = false
    })
    
    process.on('SIGINT', () => {
      logger.info('Received SIGINT, shutting down gracefully')
      this.running = false
    })
    
    process.on('SIGUSR1', () => {
      // 重新加载配置
      this.reloadConfig()
    })
    
    process.on('SIGUSR2', () => {
      // 刷新日志
      logger.flush()
    })
  }
  
  // 优雅关闭
  async shutdown(): Promise<void> {
    logger.info('Shutting down...')
    
    // 1. 停止接收新指令
    this.running = false
    
    // 2. 等待当前任务完成
    await this.waitForCurrentTasks()
    
    // 3. 保存状态
    await this.saveState()
    
    // 4. 断开连接
    await this.disconnect()
    
    logger.info('Shutdown complete')
  }
}
```

```typescript
const HEARTBEAT_CONFIG = {
  interval: 60000,        // 60 秒
  timeout: 300000,        // 300 秒 (5 分钟)
  maxMissed: 3,           // 最多允许错过 3 次

// 心跳发送
async function sendHeartbeat(sessionId: string, token: string): Promise<void> {
  try {
    const response = await fetch(`${API_BASE}/api/v1/sessions/${sessionId}/heartbeat`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${token}`,
      },
    })
    
    if (!response.ok) {
      throw new Error(`Heartbeat failed: ${response.status}`)
    }
    
    const data = await response.json()
    
    if (!data.lease_extended) {
      logger.warn('Heartbeat lease not extended')
    }
    
  } catch (error) {
    logger.error('Heartbeat failed:', error)
    throw error  // 触发重连
  }
}

// 心跳监控
class HeartbeatMonitor {
  private lastSuccess: number = Date.now()
  private missedCount: number = 0
  
  start(): void {
    setInterval(() => this.check(), HEARTBEAT_CONFIG.interval)
  }
  
  private async check(): Promise<void> {
    try {
      await sendHeartbeat()
      this.lastSuccess = Date.now()
      this.missedCount = 0
    } catch (error) {
      this.missedCount++
      
      if (this.missedCount >= HEARTBEAT_CONFIG.maxMissed) {
        logger.error('Too many missed heartbeats, reconnecting...')
        this.triggerReconnect()
      }
    }
  }
  
  private triggerReconnect(): void {
    // 触发重连逻辑
    eventEmitter.emit('reconnect')
  }
}
```

```typescript
// 状态快照
interface AgentState {
  // 任务状态
  tasks: Map<string, Task>
  activeTaskId?: string
  
  // 工人状态
  workers: Map<string, WorkerStatus>
  
  // 记忆
  memories: Memory[]
  
  // 配置
  config: AgentConfig
  
  // 统计
  stats: {
    totalTasks: number
    completedTasks: number
    failedTasks: number
    totalTokenUsage: number
  }
  
  // 最后更新时间
  lastUpdated: number
}

// 持久化
class StateManager {
  private state: AgentState
  private stateDir: string
  private saveInterval: number
  
  constructor(stateDir: string, saveIntervalMs = 60000) {
    this.stateDir = stateDir
    this.saveInterval = saveIntervalMs
    this.startAutoSave()
  }
  
  // 加载状态
  async loadState(): Promise<AgentState | null> {
    const statePath = this.getStatePath()
    
    try {
      const content = await read(statePath, 'utf-8')
      this.state = JSON.parse(content)
      return this.state
    } catch (error) {
      logger.warn('No state file found or invalid state')
      return null
    }
  }
  
  // 保存状态
  async saveState(): Promise<void> {
    const statePath = this.getStatePath()
    const tempPath = statePath + '.tmp'
    
    this.state.lastUpdated = Date.now()
    
    // 写入临时文件
    await write(tempPath, JSON.stringify(this.state, null, 2))
    
    // 原子替换
    await rename(tempPath, statePath)
  }
  
  // 自动保存
  private startAutoSave(): void {
    setInterval(() => {
      this.saveState().catch(err => {
        logger.error('Auto-save failed:', err)
      })
    }, this.saveInterval)
  }
  
  // 获取状态路径
  private getStatePath(): string {
    return join(this.stateDir, 'agent-state.json')
  }
  
  // 更新状态
  update(f: (prev: AgentState) => AgentState): void {
    this.state = f(this.state)
  }
}
```

### 3.3 状态恢复

```typescript
// 启动时恢复状态
async function restoreState(): Promise<RestorationResult> {
  const stateManager = new StateManager(STATE_DIR)
  const state = await stateManager.loadState()
  
  if (!state) {
    return { restored: false, reason: 'no_state_file' }
  }
  
  // 验证状态有效性
  const validation = validateState(state)
  if (!validation.valid) {
    return { restored: false, reason: 'invalid_state', errors: validation.errors }
  }
  
  // 恢复任务状态
  for (const [taskId, task] of Object.entries(state.tasks)) {
    if (task.status === 'in_progress') {
      // 进行中的任务需要特殊处理
      await handleInterruptedTask(task)
    }
  }
  
  // 恢复工人状态
  for (const [workerId, workerStatus] of Object.entries(state.workers)) {
    if (workerStatus.status === 'busy') {
      // 忙碌的工人需要检查是否还存活
      const alive = await checkWorkerAlive(workerId)
      if (!alive) {
        // 工人已死，任务需要重新分配
        await reassignTask(workerStatus.currentTask)
      }
    }
  }
  
  return { restored: true, state }
}

// 处理中断的任务
async function handleInterruptedTask(task: Task): Promise<void> {
  logger.info(`Handling interrupted task: ${task.id}`)
  
  // 检查任务是否有输出文件
  if (task.outputFile) {
    try {
      const output = await read(task.outputFile, 'utf-8')
      
      if (output.length > 0) {
        // 有输出，可能已经完成
        logger.info(`Task ${task.id} has output, marking as completed`)
        await updateTaskStatus(task.id, 'completed')
        return
      }
    } catch {
      // 输出文件不存在或不可读
    }
  }
  
  // 没有输出，标记为待重试
  logger.info(`Task ${task.id} has no output, marking for retry`)
  await updateTaskStatus(task.id, 'pending')
}
```

### 4.1 待命状态

```typescript
// 待命状态
type StandbyMode = 
  | 'active'        // 活跃，正常轮询
  | 'idle'          // 空闲，降低轮询频率
  | 'sleep'         // 睡眠，仅监听唤醒信号
  | 'offline'       // 离线，不响应

// 待命管理
class StandbyManager {
  private mode: StandbyMode = 'active'
  private idleTimeout: NodeJS.Timeout | null = null
  private readonly IDLE_THRESHOLD_MS = 30 * 60 * 1000  // 30 分钟无活动
  
  // 进入待命
  enterStandby(): void {
    logger.info('Entering standby mode')
    
    // 取消空闲定时器
    if (this.idleTimeout) {
      clearTimeout(this.idleTimeout)
    }
    
    // 根据当前活动决定模式
    if (this.getActiveTaskCount() === 0) {
      this.mode = 'idle'
      this.scheduleDeepSleep()
    }
  }
  
  // 唤醒
  async wakeUp(): Promise<void> {
    logger.info('Waking up from standby')
    
    this.mode = 'active'
    
    // 取消深度睡眠计划
    if (this.idleTimeout) {
      clearTimeout(this.idleTimeout)
      this.idleTimeout = null
    }
    
    // 恢复轮询
    this.resumePolling()
  }
  
  // 计划深度睡眠
  private scheduleDeepSleep(): void {
    this.idleTimeout = setTimeout(() => {
      this.mode = 'sleep'
      logger.info('Entering deep sleep')
      // 睡眠模式下只监听唤醒信号
    }, this.IDLE_THRESHOLD_MS)
  }
  
  // 检查是否有活动
  checkActivity(): void {
    if (this.getActiveTaskCount() > 0) {
      this.wakeUp()
    } else {
      this.enterStandby()
    }
  }
}
```

### 4.2 唤醒机制

```typescript
// 唤醒源
type WakeSource = 
  | 'user_message'    // 用户消息
  | 'scheduled_task'  // 计划任务
  | 'external_event'  // 外部事件
  | 'manual'          // 手动唤醒

// 唤醒处理
class WakeHandler {
  async handleWake(source: WakeSource, payload: unknown): Promise<void> {
    logger.info(`Wake from ${source}`)
    
    await standbyManager.wakeUp()
    
    switch (source) {
      case 'user_message':
        await this.handleUserMessage(payload as UserMessage)
        break
      case 'scheduled_task':
        await this.handleScheduledTask(payload as ScheduledTask)
        break
      case 'external_event':
        await this.handleExternalEvent(payload as ExternalEvent)
        break
      case 'manual':
        logger.info('Manual wake, no specific action')
        break
    }
  }
  
  private async handleUserMessage(message: UserMessage): Promise<void> {
    // 创建新任务
    const task = await taskManager.createTask({
      subject: '响应用户消息',
      description: message.content,
      priority: 'high',
    })
    
    // 立即启动
    await taskManager.startTask(task.id)
  }
  
  private async handleScheduledTask(scheduled: ScheduledTask): Promise<void> {
    // 执行计划任务
    const task = await taskManager.createTask({
      subject: scheduled.subject,
      description: scheduled.description,
      priority: scheduled.priority,
    })
    
    await taskManager.startTask(task.id)
  }
  
  private async handleExternalEvent(event: ExternalEvent): Promise<void> {
    // 处理外部事件 (如 webhook、文件变化等)
    logger.info(`External event: ${event.type}`)
    
    // 根据事件类型创建相应任务
    const task = await this.createEventTask(event)
    if (task) {
      await taskManager.startTask(task.id)
    }
  }
}
```

```typescript
// WebSocket 服务器
class WebSocketServer {
  private wss: WebSocket.Server
  private clients: Set<WebSocket> = new Set()
  
  constructor(port: number) {
    this.wss = new WebSocket.Server({ port })
    this.setupHandlers()
  }
  
  private setupHandlers(): void {
    this.wss.on('connection', (ws) => {
      this.clients.add(ws)
      
      ws.on('message', async (data) => {
        const message = JSON.parse(data.toString())
        await this.handleMessage(message, ws)
      })
      
      ws.on('close', () => {
        this.clients.delete(ws)
      })
    })
  }
  
  private async handleMessage(
    message: WebSocketMessage,
    ws: WebSocket
  ): Promise<void> {
    switch (message.type) {
      case 'user_message':
        await this.handleUserMessage(message.content, ws)
        break
      case 'task_status_query':
        await this.handleTaskStatusQuery(message.taskId, ws)
        break
      case 'cancel_task':
        await this.handleCancelTask(message.taskId, ws)
        break
    }
  }
  
  // 广播消息给所有客户端
  broadcast(message: WebSocketMessage): void {
    const data = JSON.stringify(message)
    for (const client of this.clients) {
      if (client.readyState === WebSocket.OPEN) {
        client.send(data)
      }
    }
  }
  
  // 发送任务状态更新
  notifyTaskUpdate(taskId: string, status: TaskStatus): void {
    this.broadcast({
      type: 'task_update',
      taskId,
      status,
      timestamp: Date.now(),
    })
  }
}
```

### 6.2 Token 预算管理

```typescript
// Token 预算
class TokenBudget {
  private dailyLimit: number
  private usedToday: number = 0
  private lastReset: number = Date.now()
  
  constructor(dailyLimit: number) {
    this.dailyLimit = dailyLimit
  }
  
  // 检查是否可以使用
  canUse(tokens: number): boolean {
    this.checkReset()
    return this.usedToday + tokens <= this.dailyLimit
  }
  
  // 使用 Token
  use(tokens: number): void {
    if (!this.canUse(tokens)) {
      throw new Error('Token budget exceeded')
    }
    this.usedToday += tokens
  }
  
  // 检查是否需要重置
  private checkReset(): void {
    const now = Date.now()
    const daysSinceReset = Math.floor(
      (now - this.lastReset) / (24 * 60 * 60 * 1000)
    )
    
    if (daysSinceReset >= 1) {
      this.usedToday = 0
      this.lastReset = now
    }
  }
  
  // 获取剩余 Token
  getRemaining(): number {
    this.checkReset()
    return this.dailyLimit - this.usedToday
  }
}
```

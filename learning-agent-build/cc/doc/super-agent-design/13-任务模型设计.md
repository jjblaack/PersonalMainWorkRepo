# 任务模型设计

> 任务 (Task) 是超级 Agent 的基本工作单元，本设计文档详细描述任务的数据结构、生命周期、依赖关系和操作接口。

---

## 1. 任务的核心概念

### 1.1 什么是任务

```
┌─────────────────────────────────────────────────────────┐
│                      任务 (Task)                          │
├─────────────────────────────────────────────────────────┤
│                                                         │
│  定义：超级 Agent 追踪和管理工作的基本单元                  │
│                                                         │
│  特点：                                                 │
│  - 原子性：专注于单一目标                               │
│  - 可追踪：有明确的状态和进度                           │
│  - 可验证：有明确的完成标准                             │
│  - 可依赖：可以与其他任务建立依赖关系                   │
│                                                         │
│  任务 vs 目标：                                         │
│  - 目标 (Goal)：用户提出的宏观需求，如"开发登录功能"      │
│  - 任务 (Task)：目标分解后的具体步骤，如"实现登录 API"    │
│                                                         │
│  任务 vs 工人：                                         │
│  - 任务：要完成的工作                                   │
│  - 工人：执行工作的 Agent                               │
│  - 关系：一个任务分配给一个工人执行                     │
│                                                         │
└─────────────────────────────────────────────────────────┘
```

### 2.1 完整任务定义

```typescript
// 任务状态
type TaskStatus =
  | 'pending'       // 等待执行
  | 'in_progress'   // 正在执行
  | 'completed'     // 已完成
  | 'failed'        // 执行失败
  | 'killed'        // 被停止
  | 'deleted'       // 已删除

// 任务完整定义
interface Task {
  // === 标识信息 ===
  id: string              // 唯一任务 ID
  parentId?: string       // 父任务 ID (如果是分解产生)
  source: TaskSource      // 任务来源
  
  // === 基本信息 ===
  subject: string         // 任务主题 (命令式短语)
  description: string     // 详细描述
  type: TaskType          // 任务类型
  priority: TaskPriority  // 优先级
  tags?: string[]         // 标签
  
  // === 状态信息 ===
  status: TaskStatus      // 当前状态
  owner?: string          // 执行工人 ID
  blockedBy: string[]     // 被哪些任务阻塞 (依赖)
  blocks: string[]        // 阻塞哪些任务
  
  // === 时间信息 ===
  createdAt: number       // 创建时间戳
  updatedAt: number       // 最后更新时间戳
  startedAt?: number      // 开始执行时间
  endTime?: number        // 完成/失败/停止时间
  deadline?: number       // 截止时间 (可选)
  
  // === 估算和实际 ===
  estimatedDuration?: number  // 预估耗时 (毫秒)
  actualDuration?: number     // 实际耗时 (毫秒)
  
  // === 输出信息 ===
  outputFile?: string     // 输出文件路径
  output?: string         // 输出内容摘要
  metadata?: TaskMetadata // 元数据
  
  // === 通知状态 ===
  notified: boolean       // 状态变更是否已通知
  
  // === 成功标准 ===
  successCriteria?: string[]  // 完成标准列表
}

// 任务元数据
interface TaskMetadata {
  // 重试信息
  retryCount?: number         // 重试次数
  lastError?: string          // 最后错误信息
  
  // 执行信息
  toolUses?: number           // 工具使用次数
  llmCalls?: number           // LLM 调用次数
  tokenUsage?: number         // Token 使用量
  
  // 成本信息
  costUsd?: number            // 成本 (美元)
  
  // 自定义信息

}

  type?: TaskType
  priority?: TaskPriority
  blockedBy?: string[]

  metadata?: TaskMetadata
```

### 2.2 任务 ID 生成

```typescript
// 任务 ID 生成器
class TaskIdGenerator {
  private static readonly PREFIXES: Record<TaskType, string> = {
    research: 'r',
    implementation: 'i',
    test: 't',
    documentation: 'd',
    verification: 'v',
    fix: 'f',
    refactor: 'rf',
    review: 'rv',
    deployment: 'dp',
  }
  
  private static readonly ALPHABET = '0123456789abcdefghijklmnopqrstuvwxyz'
  
  // 生成任务 ID
  static generate(type: TaskType): string {
    const prefix = this.PREFIXES[type] || 'x'
    const randomPart = this.generateRandomPart(8)
    return `${prefix}${randomPart}`
  }
  
  // 生成随机部分
  private static generateRandomPart(length: number): string {
    const bytes = randomBytes(length)
    let result = ''
    for (let i = 0; i < length; i++) {
      result += this.ALPHABET[bytes[i]! % this.ALPHABET.length]
    }
    return result
  }
  
  // 解析任务 ID
  static parse(id: string): { type: TaskType, random: string } | null {
    const prefix = id.charAt(0)
    const random = id.slice(1)
    
    const type = Object.entries(this.PREFIXES).find(
      ([, p]) => p === prefix
    )?.[0] as TaskType | undefined
    
    if (!type) return null
    
    return { type, random }
  }
}

// 使用示例
const taskId = TaskIdGenerator.generate('implementation')  // e.g., "iab3d9f2x"
const parsed = TaskIdGenerator.parse(taskId)  // { type: 'implementation', random: 'ab3d9f2x' }
```

## 3. 任务状态机

### 3.1 状态转换图

```
                          ┌─────────────────────────────────────────┐
                          │                  pending                 │
                          │  (等待执行，依赖完成后自动转为 in_progress) │
                          └────────────────┬────────────────────────┘
                                           │
                                           │ start()
                                           ▼
                          ┌─────────────────────────────────────────┐
                     ┌────│               in_progress                │────┐
                     │    │  (正在执行，工人已分配，任务进行中)       │    │
                     │    └────────────────┬────────────────────────┘    │
                     │                     │                              │
          retry()    │                     │complete()           fail()   │
                     │                     │                              │
                     │                     ▼                              │
                     │    ┌─────────────────────────────────────────┐    │
                     │    │                completed                 │    │
                     │    │  (已完成，验收通过，输出已记录)          │    │
                     │    └────────────────┬────────────────────────┘    │
                     │                     │                              │
                     │                     │reopen()                      │
                     │                     │                              │
                     │    ┌────────────────▼────────────────────────┐    │
                     │    │                deleted                   │    │
                     │    │  (已删除，软删除，数据仍保留)            │    │
                     │    └─────────────────────────────────────────┘    │
                     │                                                    │
                     └─────────────── killed ◄────────────────────────────┘
                                     (被停止，用户或系统主动停止)
```

### 3.2 状态转换规则

```typescript
// 状态转换矩阵
const VALID_TRANSITIONS: Record<TaskStatus, TaskStatus[]> = {
  'pending': ['in_progress', 'deleted'],
  'in_progress': ['completed', 'failed', 'killed', 'pending'],
  'completed': ['pending', 'deleted'],  // 可重新打开
  'failed': ['pending', 'deleted'],     // 可重试
  'killed': ['pending', 'deleted'],     // 可重新启动
  'deleted': [],  // 终端状态，不可转换
}

// 检查转换是否有效
function isValidTransition(from: TaskStatus, to: TaskStatus): boolean {
  return VALID_TRANSITIONS[from]?.includes(to) ?? false
}

// 任务状态管理器
class TaskStatusManager {
  // 更新任务状态
  async updateStatus(
    taskId: string,
    newStatus: TaskStatus,
    options?: {
      reason?: string
      owner?: string
    }
  ): Promise<Task> {
    const task = await this.loadTask(taskId)
    
    if (!isValidTransition(task.status, newStatus)) {
      throw new InvalidTransitionError(
        `Cannot transition from ${task.status} to ${newStatus}`
      )
    }
    
    const updated: Task = {
      ...task,
      status: newStatus,
      updatedAt: Date.now(),
      
      // 设置时间戳
      startedAt: newStatus === 'in_progress' ? Date.now() : task.startedAt,
      endTime: this.isTerminalStatus(newStatus) ? Date.now() : task.endTime,
      
      // 设置所有者
      owner: options?.owner ?? task.owner,
    }
    
    // 持久化
    await this.saveTask(updated)
    
    // 发送事件
    eventBus.publish('task:statusChanged', {
      taskId,
      from: task.status,
      to: newStatus,
      reason: options?.reason,
    })
    
    return updated
  }
  
  // 检查是否是终端状态
  private isTerminalStatus(status: TaskStatus): boolean {
    return ['completed', 'failed', 'killed', 'deleted'].includes(status)
  }
}

// 无效转换错误
class InvalidTransitionError extends Error {
  constructor(message: string) {
    super(message)
    this.name = 'InvalidTransitionError'
  }
}
```

### 3.3 终端状态不可变

```typescript
// 终端状态守卫
const TERMINAL_STATUSES = ['completed', 'failed', 'killed', 'deleted']

function isTerminalStatus(status: TaskStatus): boolean {
  return TERMINAL_STATUSES.includes(status)
}

// 任务修改守卫
class TaskGuard {
  // 检查任务是否可以修改
  assertModifiable(task: Task): void {
    if (isTerminalStatus(task.status)) {
      throw new TaskNotModifiableError(
        `Cannot modify task ${task.id} in terminal status ${task.status}`
      )
    }
  }
  
  // 安全更新任务
  async safeUpdate(
    taskId: string,
    updater: (task: Task) => Partial<Task>
  ): Promise<Task> {
    const task = await this.loadTask(taskId)
    
    // 检查是否可修改
    this.assertModifiable(task)
    
    // 应用更新
    const updates = updater(task)
    const updated: Task = {
      ...task,
      ...updates,
      updatedAt: Date.now(),
    }
    
    return this.saveTask(updated)
  }
}

class TaskNotModifiableError extends Error {
  constructor(message: string) {
    super(message)
    this.name = 'TaskNotModifiableError'
  }
}
```

### 4.2 依赖检查

```typescript
class TaskDependencyManager {
  // 检查任务是否可以开始
  async canStart(taskId: string): Promise<boolean> {
    const task = await this.loadTask(taskId)
    
    if (!task) return false
    if (task.status !== 'pending') return false
    
    // 检查所有依赖是否完成
    for (const depId of task.blockedBy) {
      const dep = await this.loadTask(depId)
      if (!dep || dep.status !== 'completed') {
        return false
      }
    }
    
    return true
  }
  
  // 获取所有可执行的任务
  async getExecutableTasks(): Promise<Task[]> {
    const pendingTasks = await this.listTasks({ status: 'pending' })
    const executable: Task[] = []
    
    for (const task of pendingTasks) {
      if (await this.canStart(task.id)) {
        executable.push(task)
      }
    }
    
    return executable
  }
  
  // 检测循环依赖
  hasCircularDependency(taskId: string): boolean {
    const visited = new Set<string>()
    const stack = new Set<string>()
    
    const dfs = (id: string): boolean => {
      if (stack.has(id)) return true  // 检测到循环
      if (visited.has(id)) return false
      
      visited.add(id)
      stack.add(id)
      
      const task = this.taskGraph.get(id)
      if (!task) return false
      
      for (const depId of task.blockedBy) {
        if (dfs(depId)) return true
      }
      
      stack.delete(id)
      return false
    }
    
    return dfs(taskId)
  }
  
  // 添加依赖 (带循环检测)
  async addDependency(taskId: string, dependsOn: string): Promise<void> {
    // 检查是否会形成循环
    const tempGraph = new Map(this.taskGraph)
    
    // 临时添加依赖
    const task = tempGraph.get(taskId)
    if (!task) throw new TaskNotFoundError(taskId)
    
    task.blockedBy.push(dependsOn)
    
    // 检测循环
    if (this.hasCircularDependencyWithGraph(taskId, tempGraph)) {
      throw new CircularDependencyError(
        `Adding dependency would create circular dependency`
      )
    }
    
    // 实际添加
    await this.saveTask(task)
    
    // 更新被依赖任务的 blocks
    const depTask = await this.loadTask(dependsOn)
    if (depTask) {
      depTask.blocks.push(taskId)
      await this.saveTask(depTask)
    }
  }
  
  // 移除依赖
  async removeDependency(taskId: string, dependsOn: string): Promise<void> {
    const task = await this.loadTask(taskId)
    if (!task) throw new TaskNotFoundError(taskId)
    
    task.blockedBy = task.blockedBy.filter(id => id !== dependsOn)
    await this.saveTask(task)
    
    const depTask = await this.loadTask(dependsOn)
    if (depTask) {
      depTask.blocks = depTask.blocks.filter(id => id !== taskId)
      await this.saveTask(depTask)
    }
  }
  
  // 使用临时图检测循环
  private hasCircularDependencyWithGraph(
    taskId: string,
    graph: Map<string, Task>
  ): boolean {
    const visited = new Set<string>()
    const stack = new Set<string>()
    
    const dfs = (id: string): boolean => {
      if (stack.has(id)) return true
      if (visited.has(id)) return false
      
      visited.add(id)
      stack.add(id)
      
      const task = graph.get(id)
      for (const depId of task?.blockedBy || []) {
        if (dfs(depId)) return true
      }
      
      stack.delete(id)
      return false
    }
    
    return dfs(taskId)
  }
}

class CircularDependencyError extends Error {
  constructor(message: string) {
    super(message)
    this.name = 'CircularDependencyError'
  }
}

class TaskNotFoundError extends Error {
  constructor(taskId: string) {
    super(`Task ${taskId} not found`)
    this.name = 'TaskNotFoundError'
  }
}
```

### 4.3 依赖事件传播

```typescript
// 依赖完成事件处理器
class DependencyEventHandler {
  // 当任务完成时，检查并启动依赖它的任务
  async handleTaskCompleted(completedTaskId: string): Promise<void> {
    const completedTask = await this.loadTask(completedTaskId)
    if (!completedTask) return
    
    // 获取所有被此任务阻塞的任务
    for (const blockedTaskId of completedTask.blocks) {
      const blockedTask = await this.loadTask(blockedTaskId)
      if (!blockedTask) continue
      
      // 检查是否所有依赖都完成了
      if (await this.allDependenciesCompleted(blockedTask)) {
        // 自动启动
        await this.taskManager.startTask(blockedTaskId)
        
        // 发送通知
        eventBus.publish('task:unblocked', {
          taskId: blockedTaskId,
          unblockedBy: completedTaskId,
        })
      }
    }
  }
  
  // 检查所有依赖是否完成
  private async allDependenciesCompleted(task: Task): Promise<boolean> {
    for (const depId of task.blockedBy) {
      const dep = await this.loadTask(depId)
      if (!dep || dep.status !== 'completed') {
        return false
      }
    }
    return true
  }
}
```

## 5. 任务操作接口

### 5.1 任务管理器

```typescript
class TaskManager {
  private tasks: Map<string, Task>
  private storage: TaskStorage
  
  // === 创建和删除 ===
  
  async createTask(spec: TaskSpec): Promise<Task> {
    const task: Task = {
      id: TaskIdGenerator.generate(spec.type || 'implementation'),
      source: 'human',
      subject: spec.subject,
      description: spec.description,
      type: spec.type || 'implementation',
      priority: spec.priority || 'normal',
      status: 'pending',
      blockedBy: spec.blockedBy || [],
      blocks: [],
      successCriteria: spec.successCriteria,
      estimatedDuration: spec.estimatedDuration,
      deadline: spec.deadline,
      metadata: spec.metadata,
      createdAt: Date.now(),
      updatedAt: Date.now(),
      notified: false,
    }
    
    // 验证：检查循环依赖
    if (task.blockedBy.length > 0) {
      const hasCycle = this.hasCircularDependency(task.id, task.blockedBy)
      if (hasCycle) {
        throw new CircularDependencyError(
          'Task creation would create circular dependency'
        )
      }
    }
    
    await this.storage.save(task)
    this.tasks.set(task.id, task)
    
    eventBus.publish('task:created', { task })
    
    return task
  }
  
  async deleteTask(taskId: string): Promise<void> {
    const task = await this.loadTask(taskId)
    if (!task) throw new TaskNotFoundError(taskId)
    
    // 软删除：设置为 deleted 状态
    await this.updateStatus(taskId, 'deleted')
    
    eventBus.publish('task:deleted', { taskId })
  }
  
  // === 查询 ===
  
  async getTask(taskId: string): Promise<Task | null> {
    return this.tasks.get(taskId) || await this.storage.load(taskId)
  }
  
  async listTasks(filter?: TaskFilter): Promise<Task[]> {
    let tasks = Array.from(this.tasks.values())
    
    if (filter?.status) {
      tasks = tasks.filter(t => t.status === filter.status)
    }
    
    if (filter?.type) {
      tasks = tasks.filter(t => t.type === filter.type)
    }
    
    if (filter?.priority) {
      tasks = tasks.filter(t => t.priority === filter.priority)
    }
    
    if (filter?.owner) {
      tasks = tasks.filter(t => t.owner === filter.owner)
    }
    
    // 按创建时间排序
    return tasks.sort((a, b) => b.createdAt - a.createdAt)
  }
  
  // === 状态更新 ===
  
  async updateStatus(
    taskId: string,
    status: TaskStatus,
    options?: UpdateOptions
  ): Promise<Task> {
    return this.statusManager.updateStatus(taskId, status, options)
  }
  
  async updateTask(
    taskId: string,
    updater: (task: Task) => Partial<Task>
  ): Promise<Task> {
    return this.guard.safeUpdate(taskId, updater)
  }
  
  // === 任务执行 ===
  
  async startTask(taskId: string): Promise<void> {
    const task = await this.getTask(taskId)
    if (!task) throw new TaskNotFoundError(taskId)
    
    if (!await this.dependencyManager.canStart(taskId)) {
      throw new TaskNotReadyError(`Task ${taskId} dependencies not met`)
    }
    
    await this.updateStatus(taskId, 'in_progress')
  }
  
  async completeTask(taskId: string, output?: string): Promise<void> {
    const task = await this.getTask(taskId)
    if (!task) throw new TaskNotFoundError(taskId)
    
    const updates: Partial<Task> = {
      status: 'completed',
      endTime: Date.now(),
      notified: false,
    }
    
    if (output) {
      updates.output = output
      updates.actualDuration = Date.now() - (task.startedAt || task.createdAt)
    }
    
    await this.saveUpdates(taskId, updates)
    
    // 触发依赖任务启动
    await this.dependencyManager.handleTaskCompleted(taskId)
  }
  
  async failTask(taskId: string, error: string): Promise<void> {
    const task = await this.getTask(taskId)
    if (!task) throw new TaskNotFoundError(taskId)
    
    await this.saveUpdates(taskId, {
      status: 'failed',
      endTime: Date.now(),
      metadata: {
        ...task.metadata,
        lastError: error,
      },
      notified: false,
    })
  }
  
  async killTask(taskId: string): Promise<void> {
    const task = await this.getTask(taskId)
    if (!task) throw new TaskNotFoundError(taskId)
    
    // 停止关联的工人
    if (task.owner) {
      await this.workerPool.killWorker(task.owner)
    }
    
    await this.updateStatus(taskId, 'killed')
  }
  
  // === 重试 ===
  
  async retryTask(taskId: string): Promise<void> {
    const task = await this.getTask(taskId)
    if (!task) throw new TaskNotFoundError(taskId)
    
    if (task.status !== 'failed') {
      throw new InvalidOperationError(
        `Can only retry failed tasks, got ${task.status}`
      )
    }
    
    const retryCount = task.metadata?.retryCount || 0
    if (retryCount >= MAX_RETRIES) {
      throw new MaxRetriesExceededError(
        `Task ${taskId} exceeded maximum retries (${MAX_RETRIES})`
      )
    }
    
    await this.saveUpdates(taskId, {
      status: 'pending',
      owner: undefined,
      metadata: {
        ...task.metadata,
        retryCount: retryCount + 1,
      },
      notified: false,
    })
  }
  
  // === 辅助方法 ===
  
  private async saveUpdates(taskId: string, updates: Partial<Task>): Promise<void> {
    const task = await this.getTask(taskId)
    if (!task) throw new TaskNotFoundError(taskId)
    
    const updated: Task = {
      ...task,
      ...updates,
      updatedAt: Date.now(),
    }
    
    await this.storage.save(updated)
    this.tasks.set(taskId, updated)
    
    eventBus.publish('task:updated', { task: updated })
  }
}

interface TaskFilter {
  status?: TaskStatus
  type?: TaskType
  priority?: TaskPriority
  owner?: string
}

interface UpdateOptions {
  reason?: string
  owner?: string
}

class TaskNotReadyError extends Error {
  constructor(message: string) {
    super(message)
    this.name = 'TaskNotReadyError'
  }
}

class InvalidOperationError extends Error {
  constructor(message: string) {
    super(message)
    this.name = 'InvalidOperationError'
  }
}

  constructor(message: string) {
```

```typescript
// 任务存储接口
interface TaskStorage {

// JSONL 存储实现
class JsonlTaskStorage implements TaskStorage {
  private filePath: string
  private index: Map<string, number> = new Map()  // taskId -> line number
  
  constructor(filePath: string) {
    this.filePath = filePath
  }
  
  // 保存任务 (追加模式)
  async save(task: Task): Promise<void> {
    const entry = {
      ...task,
      _timestamp: Date.now(),
    }
    
    await appendFile(this.filePath, JSON.stringify(entry) + '\n')
  }
  
  // 加载任务 (从快照和日志重建)
  async load(taskId: string): Promise<Task | null> {
    const content = await read(this.filePath, 'utf-8')
    const lines = content.split('\n').filter(l => l.trim())
    
    // 从日志中重建任务状态
    let task: Task | null = null
    
    for (const line of lines) {
      const entry = JSON.parse(line)
      if (entry.id === taskId) {
        task = { ...entry }
      }
    }
    
    return task
  }
  
  // 删除任务 (软删除，不实际移除)
  async delete(taskId: string): Promise<void> {
    // 软删除通过更新状态实现，不需要实际删除记录
  }
  
  // 列出任务
  async list(filter?: TaskFilter): Promise<Task[]> {
    const content = await read(this.filePath, 'utf-8')
    const lines = content.split('\n').filter(l => l.trim())
    
    const tasks: Task[] = []
    const latestTasks = new Map<string, Task>()
    
    // 读取所有记录，保留每个任务的最新状态
    for (const line of lines) {
      const entry = JSON.parse(line)
      latestTasks.set(entry.id, { ...entry })
    }
    
    // 转换为数组并过滤
    for (const task of latestTasks.values()) {
      if (task.status === 'deleted') continue
      
      if (filter?.status && task.status !== filter.status) continue
      if (filter?.type && task.type !== filter.type) continue
      
      tasks.push(task)
    }
    
    return tasks.sort((a, b) => b.createdAt - a.createdAt)
  }
  
  // 创建快照 (压缩日志)
  async createSnapshot(): Promise<void> {
    const tasks = await this.list()
    const snapshotPath = this.filePath.replace('.jsonl', '-snapshot.json')
    
    await write(snapshotPath, JSON.stringify({
      timestamp: Date.now(),
      tasks,
    }, null, 2))
  }
  
  // 从快照加载
  async loadFromSnapshot(): Promise<Map<string, Task>> {
    const snapshotPath = this.filePath.replace('.jsonl', '-snapshot.json')
    
    try {
      const content = await read(snapshotPath, 'utf-8')
      const { tasks } = JSON.parse(content)
      
      return new Map(tasks.map((t: Task) => [t.id, t]))
    } catch {
      return new Map()
    }
  }
}
```

```typescript
class TaskQuery {
  // 获取任务统计
  async getStats(): Promise<TaskStats> {
    const tasks = await this.taskManager.listTasks()
    
    return {
      total: tasks.length,
      byStatus: {
        pending: tasks.filter(t => t.status === 'pending').length,
        in_progress: tasks.filter(t => t.status === 'in_progress').length,
        completed: tasks.filter(t => t.status === 'completed').length,
        failed: tasks.filter(t => t.status === 'failed').length,
        killed: tasks.filter(t => t.status === 'killed').length,
      },
      byType: this.groupByType(tasks),
      byPriority: this.groupByPriority(tasks),
    }
  }
  
  // 获取进行中的任务
  async getActiveTasks(): Promise<Task[]> {
    return this.taskManager.listTasks({ status: 'in_progress' })
  }
  
  // 获取待处理的任务
  async getPendingTasks(): Promise<Task[]> {
    return this.taskManager.listTasks({ status: 'pending' })
  }
  
  // 获取可执行的任务
  async getExecutableTasks(): Promise<Task[]> {
    const pending = await this.getPendingTasks()
    const executable: Task[] = []
    
    for (const task of pending) {
      if (await this.dependencyManager.canStart(task.id)) {
        executable.push(task)
      }
    }
    
    return executable
  }
  
  // 获取失败的任务
  async getFailedTasks(): Promise<Task[]> {
    return this.taskManager.listTasks({ status: 'failed' })
  }
  
  // 搜索任务
  async search(query: string): Promise<Task[]> {
    const tasks = await this.taskManager.listTasks()
    const lowerQuery = query.toLowerCase()
    
    return tasks.filter(task =>
      task.subject.toLowerCase().includes(lowerQuery) ||
      task.description.toLowerCase().includes(lowerQuery) ||
      task.tags?.some(tag => tag.toLowerCase().includes(lowerQuery))
    )
  }
  
  private groupByType(tasks: Task[]): Record<TaskType, number> {
    const result: Partial<Record<TaskType, number>> = {}
    for (const task of tasks) {
      result[task.type] = (result[task.type] || 0) + 1
    }
    return result as Record<TaskType, number>
  }
  
  private groupByPriority(tasks: Task[]): Record<TaskPriority, number> {
    const result: Partial<Record<TaskPriority, number>> = {}
    for (const task of tasks) {
      result[task.priority] = (result[task.priority] || 0) + 1
    }
    return result as Record<TaskPriority, number>
  }
}
```

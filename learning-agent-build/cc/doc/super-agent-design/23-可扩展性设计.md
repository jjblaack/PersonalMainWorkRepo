## 1. 可扩展性挑战

### 1.1 主要挑战

| 挑战 | 说明 | 影响 |
|------|------|------|
| 工人数量增长 | 从几个工人到上百个工人 | 管理复杂度指数增长 |
| 任务量增长 | 从每天几个任务到几千个任务 | 调度和存储压力 |
| 功能扩展 | 新增工人类型、工具、记忆类型 | 代码维护和升级 |
| 多实例部署 | 多个 Agent 实例协同工作 | 状态同步和任务分配 |

### 2.2 多池架构设计

```typescript
// 多工人池架构
class MultiWorkerPoolSystem {
  // 工人池分片
  private pools: Map<string, WorkerPool> = new Map()
  
  // 池类型
  private readonly POOL_TYPES = {
    'research': { minWorkers: 1, maxWorkers: 5 },
    'implementation': { minWorkers: 2, maxWorkers: 10 },
    'verification': { minWorkers: 1, maxWorkers: 3 },
    'general': { minWorkers: 1, maxWorkers: 5 },
  }
  
  // 获取或创建工人 (按类型)
  async acquireWorker(taskType: string): Promise<Worker> {
    // 1. 根据任务类型选择池
    const poolKey = this.getPoolKey(taskType)
    
    // 2. 从对应池获取工人
    const pool = this.getOrCreatePool(poolKey)
    return await pool.acquireWorker()
  }
  
  private getPoolKey(taskType: string): string {
    // 实现类型 → 池映射
    const mapping: Record<string, string> = {
      'research': 'research',
      'analysis': 'research',
      'implementation': 'implementation',
      'refactor': 'implementation',
      'test': 'verification',
      'review': 'verification',
    }
    return mapping[taskType] || 'general'
  }
  
  private getOrCreatePool(poolKey: string): WorkerPool {
    if (!this.pools.has(poolKey)) {
      const config = this.POOL_TYPES[poolKey] || this.POOL_TYPES['general']
      this.pools.set(poolKey, new WorkerPool(config))
    }
    return this.pools.get(poolKey)!
  }
}
```

### 2.3 动态工人池扩展

```typescript
// 动态扩展的工人池
class ScalableWorkerPool {
  private workers: Worker[] = []
  private minWorkers: number
  private maxWorkers: number
  private scaleUpThreshold: number  // 使用率超过此值触发扩容
  private scaleDownThreshold: number  // 使用率低于此值触发缩容
  
  // 自动扩缩容
  async autoScale(): Promise<void> {
    const utilization = this.getUtilization()
    
    if (utilization > this.scaleUpThreshold && this.workers.length < this.maxWorkers) {
      await this.scaleUp()
    } else if (utilization < this.scaleDownThreshold && this.workers.length > this.minWorkers) {
      await this.scaleDown()
    }
  }
  
  // 扩容
  private async scaleUp(): Promise<void> {
    const targetSize = Math.min(
      this.workers.length + 2,  // 每次增加 2 个
      this.maxWorkers
    )
    
    logger.info(`Scaling up worker pool: ${this.workers.length} → ${targetSize}`)
    
    while (this.workers.length < targetSize) {
      const worker = await this.createWorker()
      this.workers.push(worker)
    }
  }
  
  // 缩容
  private async scaleDown(): Promise<void> {
    const targetSize = Math.max(
      this.workers.length - 1,  // 每次减少 1 个
      this.minWorkers
    )
    
    // 只清理空闲工人
    const idleWorkers = this.workers.filter(w => w.status === 'idle')
    const toRemove = idleWorkers.slice(0, this.workers.length - targetSize)
    
    logger.info(`Scaling down worker pool: ${this.workers.length} → ${targetSize}`)
    
    for (const worker of toRemove) {
      await worker.cleanup()
      this.workers = this.workers.filter(w => w !== worker)
    }
  }
  
  // 计算使用率
  private getUtilization(): number {
    const busyCount = this.workers.filter(w => w.status === 'busy').length
    return busyCount / this.workers.length
  }
}
```

### 2.4 工人池分片策略

```typescript
// 基于一致性哈希的工人池分片
class ConsistentHashSharding {
  private readonly VIRTUAL_NODES = 150  // 每个物理节点的虚拟节点数
  private ring: SortedMap<number, string> = new SortedMap()  // hash → poolId
  
  // 添加池
  addPool(poolId: string): void {
    for (let i = 0; i < this.VIRTUAL_NODES; i++) {
      const hash = this.hash(`${poolId}-${i}`)
      this.ring.set(hash, poolId)
    }
  }
  
  // 移除池
  removePool(poolId: string): void {
    for (let i = 0; i < this.VIRTUAL_NODES; i++) {
      const hash = this.hash(`${poolId}-${i}`)
      this.ring.delete(hash)
    }
  }
  
  // 获取任务对应的池
  getPoolForTask(taskId: string): string | null {
    if (this.ring.size === 0) return null
    
    const hash = this.hash(taskId)
    
    // 查找顺时针第一个节点
    for (const [ringHash, poolId] of this.ring.entries()) {
      if (ringHash >= hash) {
        return poolId
      }
    }
    
    // 回到环的开头
    return this.ring.values().next().value
  }
  
  private hash(key: string): number {
    // 使用 MD5 或 murmurhash
    return murmurhash3_32_gc(key, 0x12345678)
  }
}

// murmurhash3 实现 (简化版)
function murmurhash3_32_gc(key: string, seed: number): number {
  let h1 = seed
  const len = key.length
  
  for (let i = 0; i < len; i++) {
    let k1 = key.charCodeAt(i)
    k1 = Math.imul(k1, 0xcc9e2d51)
    k1 = (k1 << 15) | (k1 >>> 17)
    k1 = Math.imul(k1, 0x1b873593)
    h1 ^= k1
    h1 = (h1 << 13) | (h1 >>> 19)
    h1 = Math.imul(h1, 5) + 0xe6546b64
  }
  
  h1 ^= len
  h1 ^= h1 >>> 16
  h1 = Math.imul(h1, 0x85ebca6b)
  h1 ^= h1 >>> 13
  h1 = Math.imul(h1, 0xc2b2ae35)
  h1 ^= h1 >>> 16
  
  return h1 >>> 0  // 转为无符号整数
}
```

### 3.1 任务存储分片

```typescript
// 任务存储分片
class ShardedTaskStorage {
  private readonly SHARD_COUNT = 16
  private shards: Map<number, JsonlTaskStorage> = new Map()
  
  constructor(baseDir: string) {
    for (let i = 0; i < this.SHARD_COUNT; i++) {
      const shardDir = join(baseDir, `shard-${i}`)
      this.shards.set(i, new JsonlTaskStorage(shardDir))
    }
  }
  
  // 根据任务 ID 计算分片
  private getShardId(taskId: string): number {
    const hash = this.hashTaskId(taskId)
    return hash % this.SHARD_COUNT
  }
  
  private hashTaskId(taskId: string): number {
    let hash = 0
    for (let i = 0; i < taskId.length; i++) {
      hash = ((hash << 5) - hash) + taskId.charCodeAt(i)
      hash |= 0
    }
    return Math.abs(hash)
  }
  
  // CRUD 操作
  async saveTask(task: Task): Promise<void> {
    const shard = this.shards.get(this.getShardId(task.id))!
    await shard.saveTask(task)
  }
  
  async getTask(taskId: string): Promise<Task | null> {
    const shard = this.shards.get(this.getShardId(taskId))!
    return await shard.getTask(taskId)
  }
  
  async updateTask(taskId: string, update: Partial<Task>): Promise<void> {
    const shard = this.shards.get(this.getShardId(taskId))!
    await shard.updateTask(taskId, update)
  }
}
```

### 3.2 任务调度分片

```typescript
// 任务调度器分片
class ShardedTaskScheduler {
  private readonly SHARD_COUNT = 8
  private shards: Map<number, TaskShard> = new Map()
  
  constructor() {
    for (let i = 0; i < this.SHARD_COUNT; i++) {
      this.shards.set(i, new TaskShard(i))
    }
  }
  
  // 创建任务
  async createTask(task: Task): Promise<void> {
    const shard = this.getShardForTask(task.id)
    await shard.addTask(task)
  }
  
  // 获取下一个可执行任务
  async getNextExecutableTask(): Promise<Task | null> {
    // 轮询所有分片
    const promises = Array.from(this.shards.values()).map(shard => 
      shard.getExecutableTask()
    )
    
    const results = await Promise.all(promises)
    
    // 返回优先级最高的任务
    const executableTasks = results.filter((t): t is Task => t !== null)
    if (executableTasks.length === 0) return null
    
    return executableTasks.sort((a, b) => 
      b.priority - a.priority || a.createdAt - b.createdAt
    )[0]
  }
}

// 任务分片
class TaskShard {
  private tasks: Map<string, Task> = new Map()
  private priorityQueues: Map<number, string[]> = new Map()  // priority → taskIds
  
  constructor(private shardId: number) {}
  
  async addTask(task: Task): Promise<void> {
    this.tasks.set(task.id, task)
    
    const priority = task.priority || 0
    if (!this.priorityQueues.has(priority)) {
      this.priorityQueues.set(priority, [])
    }
    this.priorityQueues.get(priority)!.push(task.id)
  }
  
  async getExecutableTask(): Promise<Task | null> {
    // 按优先级从高到低查找
    const priorities = Array.from(this.priorityQueues.keys()).sort((a, b) => b - a)
    
    for (const priority of priorities) {
      const queue = this.priorityQueues.get(priority)!
      while (queue.length > 0) {
        const taskId = queue.shift()!
        const task = this.tasks.get(taskId)
        
        if (task && this.isTaskExecutable(task)) {
          return task
        }
      }
    }
    
    return null
  }
  
  private isTaskExecutable(task: Task): boolean {
    // 检查依赖是否完成
    for (const depId of task.blockedBy) {
      const dep = this.tasks.get(depId)
      if (!dep || dep.status !== 'completed') {
        return false
      }
    }
    return true
  }
}
```

```typescript
// 分布式任务依赖管理
class DistributedDependencyManager {
  // 跨分片依赖检测
  async checkCrossShardDependencies(task: Task): Promise<DependencyCheckResult> {
    const result: DependencyCheckResult = {
      canStart: true,
      blockingDeps: [],
      errors: [],
    }
    
    // 收集所有依赖任务 (可能在不同分片)
    const depTasks = await Promise.all(
      task.blockedBy.map(depId => this.getTaskFromAnyShard(depId))
    )
    
    // 检查依赖状态
    for (let i = 0; i < task.blockedBy.length; i++) {
      const depId = task.blockedBy[i]
      const depTask = depTasks[i]
      
      if (!depTask) {
        result.errors.push(`Dependency ${depId} not found`)
        result.canStart = false
      } else if (depTask.status !== 'completed') {
        result.blockingDeps.push(depId)
        result.canStart = false
      }
    }
    
    return result
  }
  
  // 从任意分片获取任务
  private async getTaskFromAnyShard(taskId: string): Promise<Task | null> {
    // 先计算分片
    const shard = this.getShardForTask(taskId)
    let task = await shard.getTask(taskId)
    
    if (!task) {
      // 跨分片查找 (广播)
      for (const [id, s] of this.shards) {
        if (id !== shard.shardId) {
          task = await s.getTask(taskId)
          if (task) break
        }
      }
    }
    
    return task
  }
}
```

### 4.3 插件加载器

```typescript
// 插件加载器
class PluginLoader {
  private plugins: Map<string, Plugin> = new Map()
  private pluginDir: string
  
  constructor(pluginDir: string) {
    this.pluginDir = pluginDir
  }
  
  // 加载所有插件
  async loadAllPlugins(): Promise<void> {
    const pluginFiles = await this.discoverPlugins()
    
    for (const file of pluginFiles) {
      try {
        await this.loadPlugin(file)
      } catch (error) {
        logger.error(`Failed to load plugin ${file}:`, error)
      }
    }
    
    // 按依赖顺序激活插件
    await this.activatePluginsInOrder()
  }
  
  // 发现插件
  private async discoverPlugins(): Promise<string[]> {
    const files = await readdir(this.pluginDir)
    return files
      .filter(f => f.endsWith('.plugin.js') || f.endsWith('.plugin.ts'))
      .map(f => join(this.pluginDir, f))
  }
  
  // 加载单个插件
  private async loadPlugin(file: string): Promise<void> {
    const module = await import(file)
    const plugin: Plugin = module.default || module.plugin
    
    // 验证插件接口
    this.validatePlugin(plugin)
    
    // 存储插件
    this.plugins.set(plugin.name, plugin)
    logger.info(`Loaded plugin: ${plugin.name} v${plugin.version}`)
  }
  
  // 验证插件接口
  private validatePlugin(plugin: Plugin): void {
    if (!plugin.name) {
      throw new Error('Plugin must have a name')
    }
    if (!plugin.version) {
      throw new Error('Plugin must have a version')
    }
    if (typeof plugin.activate !== 'function') {
      throw new Error('Plugin must have an activate function')
    }
  }
  
  // 按依赖顺序激活
  private async activatePluginsInOrder(): Promise<void> {
    const activated = new Set<string>()
    const context = this.createPluginContext()
    
    // 拓扑排序
    const sorted = this.topologicalSort()
    
    for (const pluginName of sorted) {
      const plugin = this.plugins.get(pluginName)!
      await plugin.activate(context)
      activated.add(pluginName)
      logger.info(`Activated plugin: ${pluginName}`)
    }
  }
  
  // 拓扑排序 (处理依赖)
  private topologicalSort(): string[] {
    const result: string[] = []
    const visited = new Set<string>()
    const visiting = new Set<string>()
    
    const visit = (pluginName: string) => {
      if (visited.has(pluginName)) return
      if (visiting.has(pluginName)) {
        throw new Error(`Circular dependency detected: ${pluginName}`)
      }
      
      visiting.add(pluginName)
      
      const plugin = this.plugins.get(pluginName)
      if (plugin?.dependencies) {
        for (const dep of plugin.dependencies) {
          visit(dep)
        }
      }
      
      visiting.delete(pluginName)
      visited.add(pluginName)
      result.push(pluginName)
    }
    
    for (const pluginName of this.plugins.keys()) {
      visit(pluginName)
    }
    
    return result
  }
  
  private createPluginContext(): PluginContext {
    // 创建插件上下文
    return {
      taskManager: TaskManager.getInstance(),
      workerPool: WorkerPool.getInstance(),
      memorySystem: MemorySystem.getInstance(),
      logger: logger.child({ component: 'plugin' }),
      registry: this.registry,
      config: this.config,
    }
  }
}
```

```typescript
// 配置热加载
class HotReloadableConfig {
  private config: MergedConfig
  private watchers: Set<ConfigChangeListener> = new Set()
  
  constructor(config: MergedConfig) {
    this.config = config
    this.watchConfigFile()
  }
  
  // 监听配置文件变化
  private watchConfigFile(): void {
    const configPath = join(process.cwd(), 'config.yaml')
    
    fs.watch(configPath, async (eventType) => {
      if (eventType === 'change') {
        try {
          await this.reload()
        } catch (error) {
          logger.error('Failed to reload config:', error)
        }
      }
    })
  }
  
  // 重新加载配置
  async reload(): Promise<void> {
    logger.info('Reloading configuration...')
    
    const loader = new ConfigLoader()
    const newConfig = await loader.load()
    
    // 计算变化
    const changes = this.detectChanges(this.config, newConfig)
    
    // 应用新配置
    this.config = newConfig
    
    // 通知监听者
    for (const listener of this.watchers) {
      await listener(changes)
    }
    
    logger.info(`Configuration reloaded. ${changes.length} changes detected.`)
  }
  
  // 检测变化
  private detectChanges(old: MergedConfig, new: MergedConfig): ConfigChange[] {
    const changes: ConfigChange[] = []
    
    for (const key of Object.keys(new)) {
      if (!deepEqual(old[key], new[key])) {
        changes.push({
          key,
          oldValue: old[key],
          newValue: new[key],
        })
      }
    }
    
    return changes
  }
  
  // 订阅配置变化
  onChange(listener: ConfigChangeListener): () => void {
    this.watchers.add(listener)
    return () => this.watchers.delete(listener)
  }
  
  // 获取配置
  get<T>(key: string): T {
    return (this.config as any)[key]
  }
}

// 配置变化
interface ConfigChange {
  key: string
  oldValue: unknown
  newValue: unknown
}

type ConfigChangeListener = (changes: ConfigChange[]) => Promise<void>
```

```typescript
// 并发限制器
class ConcurrencyLimiter {
  private running = 0
  private queue: Array<() => void> = []
  
  constructor(private maxConcurrency: number) {}
  
  // 执行任务 (受并发限制)
  async execute<T>(fn: () => Promise<T>): Promise<T> {
    if (this.running >= this.maxConcurrency) {
      // 等待
      await new Promise<void>(resolve => this.queue.push(resolve))
    }
    
    this.running++
    
    try {
      return await fn()
    } finally {
      this.running--
      this.next()
    }
  }
  
  private next(): void {
    const next = this.queue.shift()
    if (next) {
      next()
    }
  }
  
  // 获取状态
  getStatus(): { running: number; queued: number } {
    return {
      running: this.running,
      queued: this.queue.length,
    }
  }
}

// 使用示例
const limiter = new ConcurrencyLimiter(5)  // 最多 5 个并发

  tasks.map(task => limiter.execute(() => executeTask(task)))
```

```typescript
// 两级缓存系统
class TwoLevelCache<K, V> {
  // L1: 内存缓存 (快速，容量小)
  private l1: LRUCache<K, V>
  
  // L2: 磁盘缓存 (慢速，容量大)
  private l2: DiskCache<K, V>
  
  constructor(l1Size: number, l2Dir: string) {
    this.l1 = new LRUCache(l1Size)
    this.l2 = new DiskCache(l2Dir)
  }
  
  async get(key: K): Promise<V | null> {
    // 1. 尝试 L1
    let value = this.l1.get(key)
    if (value !== undefined) {
      return value
    }
    
    // 2. 尝试 L2
    value = await this.l2.get(key)
    if (value !== null) {
      // 回填 L1
      this.l1.set(key, value)
      return value
    }
    
    return null
  }
  
  async set(key: K, value: V): Promise<void> {
    // 同时写入 L1 和 L2
    this.l1.set(key, value)
    await this.l2.set(key, value)
  }
  
  async delete(key: K): Promise<void> {
    this.l1.delete(key)
    await this.l2.delete(key)
  }
}

// LRU 缓存实现
class LRUCache<K, V> {
  private cache: Map<K, V>
  private maxSize: number
  
  constructor(maxSize: number) {
    this.cache = new Map()
    this.maxSize = maxSize
  }
  
  get(key: K): V | undefined {
    const value = this.cache.get(key)
    if (value !== undefined) {
      // 移到末尾 (最近使用)
      this.cache.delete(key)
      this.cache.set(key, value)
    }
    return value
  }
  
  set(key: K, value: V): void {
    if (this.cache.has(key)) {
      this.cache.delete(key)
    } else if (this.cache.size >= this.maxSize) {
      // 删除最旧的
      const oldestKey = this.cache.keys().next().value
      this.cache.delete(oldestKey)
    }
    this.cache.set(key, value)
  }
  
  delete(key: K): void {
    this.cache.delete(key)
  }
}

// 磁盘缓存
class DiskCache<K, V> {
  private dir: string
  
  constructor(dir: string) {
    this.dir = dir
  }
  
  async get(key: K): Promise<V | null> {
    const filePath = this.getFilePath(key)
    try {
      const content = await readFile(filePath, 'utf-8')
      return JSON.parse(content) as V
    } catch {
      return null
    }
  }
  
  async set(key: K, value: V): Promise<void> {
    const filePath = this.getFilePath(key)
    await writeFile(filePath, JSON.stringify(value))
  }
  
  async delete(key: K): Promise<void> {
    const filePath = this.getFilePath(key)
    await unlink(filePath).catch(() => {})
  }
  
  private getFilePath(key: K): string {
    const hash = createHash('md5').update(String(key)).digest('hex')
    return join(this.dir, `${hash}.json`)
  }
}
```

```typescript
class LoadBalancer {
  private instances: Map<string, InstanceInfo> = new Map()
  private strategy: LoadBalanceStrategy
  
  constructor(strategy: LoadBalanceStrategy = 'round-robin') {
    this.strategy = strategy
  }
  
  // 注册实例
  register(instance: InstanceInfo): void {
    this.instances.set(instance.id, instance)
  }
  
  // 注销实例
  unregister(instanceId: string): void {
    this.instances.delete(instanceId)
  }
  
  // 选择实例
  selectInstance(taskId: string): InstanceInfo | null {
    const available = Array.from(this.instances.values())
      .filter(i => i.status === 'healthy' && i.available)
    
    if (available.length === 0) return null
    
    switch (this.strategy) {
      case 'round-robin':
        return this.roundRobin(available)
      case 'least-connections':
        return this.leastConnections(available)
      case 'consistent-hash':
        return this.consistentHash(available, taskId)
      default:
        return available[0]
    }
  }
  
  private roundRobin(instances: InstanceInfo[]): InstanceInfo {
    // 简单轮询
    const index = this.getIndex() % instances.length
    return instances[index]
  }
  
  private leastConnections(instances: InstanceInfo[]): InstanceInfo {
    return instances.reduce((min, inst) => 
      inst.connections < min.connections ? inst : min
    )
  }
  
  private consistentHash(instances: InstanceInfo[], taskId: string): InstanceInfo {
    const hash = murmurhash3_32_gc(taskId, 0)
    const index = hash % instances.length
    return instances[index]
  }
  
  private getIndex(): number {
    // 实现轮询索引
    return Date.now() % 1000
  }
}

  id: string
  host: string
  port: number
  status: 'healthy' | 'unhealthy' | 'draining'
  available: boolean
  connections: number
  lastHeartbeat: number
}
```

```typescript
// 基于 Redis 的分布式锁
class DistributedLock {
  private redis: Redis
  private readonly LOCK_TIMEOUT_MS = 30000
  
  constructor(redis: Redis) {
    this.redis = redis
  }
  
  // 获取锁
  async acquire(lockKey: string, ownerId: string): Promise<boolean> {
    const result = await this.redis.set(
      `lock:${lockKey}`,
      ownerId,
      'PX',
      this.LOCK_TIMEOUT_MS,
      'NX'  // 仅在键不存在时设置
    )
    return result === 'OK'
  }
  
  // 释放锁
  async release(lockKey: string, ownerId: string): Promise<void> {
    const currentValue = await this.redis.get(`lock:${lockKey}`)
    if (currentValue === ownerId) {
      await this.redis.del(`lock:${lockKey}`)
    }
  }
  
  // 带重试的获取锁
  async acquireWithRetry(
    lockKey: string,
    ownerId: string,
    maxRetries: number = 5,
    retryDelayMs: number = 100
  ): Promise<boolean> {
    for (let i = 0; i < maxRetries; i++) {
      if (await this.acquire(lockKey, ownerId)) {
        return true
      }
      await sleep(retryDelayMs * (i + 1))  // 指数退避
    }
    return false
  }
}

// 使用分布式锁的任务分配
class DistributedTaskAllocator {
  private lock: DistributedLock
  
  constructor(lock: DistributedLock) {
    this.lock = lock
  }
  
  async allocateTask(taskId: string, allocatorId: string): Promise<boolean> {
    // 获取分配锁
    const acquired = await this.lock.acquireWithRetry(
      `task-alloc:${taskId}`,
      allocatorId
    )
    
    if (!acquired) {
      logger.warn(`Failed to acquire lock for task ${taskId}`)
      return false
    }
    
    try {
      // 检查任务是否已被分配
      const existing = await this.getTaskAllocator(taskId)
      if (existing) {
        return false  // 已被其他分配器分配
      }
      
      // 分配任务
      await this.setTaskAllocator(taskId, allocatorId)
      return true
    } finally {
      await this.lock.release(`task-alloc:${taskId}`, allocatorId)
    }
  }
}
```

```typescript
      newState,
      timestamp: Date.now(),
      source: this.getInstanceId(),

    })
  }

    // 忽略自己发布的消息
    if (message.source === this.getInstanceId()) return
    
    // 更新本地状态
    this.updateLocalState(message)
  }
  
  private getInstanceId(): string {

  }
```

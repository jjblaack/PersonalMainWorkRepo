async function runHeadless(args: CLIArgs) {
  // 1. 初始化环境
  await init()
  
  // 2. 创建 QueryEngine (无 UI)
  const engine = new QueryEngine({
    cwd: getCwd(),
    tools: await getTools(),
    commands: await getCommands(),
    canUseTool: buildCanUseToolFn(),

function parseUserInput(input: string): ParsedInput {
  const trimmed = input.trim()
  
  // 检测 slash 命令
  if (trimmed.startsWith('/')) {
    const parts = trimmed.slice(1).split(/\s+/)
    const commandName = parts[0]
    const args = parts.slice(1).join(' ')
    
    return {
      type: 'slash_command',
      commandName,
      args,
      raw: input,
    }
  }
  
  // 自然语言输入
  return {
    type: 'natural_language',
    content: input,
  }
}

### 4.2 submitMessage 流程

**位置**: `src/QueryEngine.ts:209+`

```typescript
async *submitMessage(prompt, options) {
  const {
    tools, mcpClients, canUseTool,
    getAppState, setAppState,
  } = this.config
  
  // === 阶段 1: 准备 ===
  
  // 1.1 设置 CWD
  setCwd(this.config.cwd)
  
  // 1.2 构建系统提示
  const { defaultSystemPrompt, userContext, systemContext } = 
    await fetchSystemPromptParts({ tools, mcpClients })
  
  // 1.3 加载记忆
  const memoryPrompt = loadMemoryPrompt(getMemoryDir(getProjectRoot()))
  
  // 1.4 添加用户消息到历史
  const userMessage = createUserMessage(prompt)
  this.mutableMessages.push(userMessage)
  
  // === 阶段 2: LLM 调用循环 ===
  
  let turnCount = 0
  const maxTurns = this.config.maxTurns ?? 50
  
  while (turnCount < maxTurns) {
    // 2.1 发送请求到 LLM
    const response = await this.send({
      systemPrompt: defaultSystemPrompt + memoryPrompt,
      messages: this.mutableMessages,
      tools: getToolDefinitions(tools),
    })
    
    // 2.2 处理 LLM 响应
    if (response.stop_reason === 'end_turn') {
      // 对话完成
      break
    }
    
    if (response.stop_reason === 'tool_use') {
      // 2.3 执行工具调用
      const toolResults = await this.executeToolCalls(
        response.content,
        canUseTool,
        tools
      )
      
      // 2.4 添加工具结果到消息历史
      this.mutableMessages.push(...toolResults)
      
      // 2.5 继续下一次循环
      turnCount++
      continue
    }
  }
  
  // === 阶段 3: 返回结果 ===
  
  yield {
    type: 'assistant',
    content: extractAssistantContent(response),
    usage: this.totalUsage,
  }
}
```

### 4.3 工具执行循环

**位置**: `src/QueryEngine.ts` (推断)

```typescript
private async executeToolCalls(
  content: ContentBlockParam[],
  canUseTool: CanUseToolFn,
  tools: Tools
): Promise<ToolResult[]> {
  const toolUses = content.filter(block => block.type === 'tool_use')
  const results: ToolResult[] = []
  
  for (const toolUse of toolUses) {
    // 查找工具
    const tool = tools.find(t => t.name === toolUse.name)
    
    if (!tool) {
      results.push({
        type: 'tool_result',
        tool_use_id: toolUse.id,
        content: [{ type: 'text', text: `Unknown tool: ${toolUse.name}` }],
        is_error: true,
      })
      continue
    }
    
    // 权限检查
    const permissionResult = await canUseTool(
      tool,
      toolUse.input,
      this.getToolContext()
    )
    
    if (permissionResult.behavior === 'deny') {
      results.push({
        type: 'tool_result',
        tool_use_id: toolUse.id,
        content: [{ type: 'text', text: 'Permission denied' }],
        is_error: true,
      })
      continue
    }
    
    // 执行工具
    try {
      const toolResult = await tool.execute(toolUse.input, this.getToolContext())
      
      results.push({
        type: 'tool_result',
        tool_use_id: toolUse.id,
        content: toolResult.content,
        is_error: toolResult.success === false,
      })
    } catch (error) {
      results.push({
        type: 'tool_result',
        tool_use_id: toolUse.id,
        content: [{ type: 'text', text: errorMessage(error) }],
        is_error: true,
      })
    }
  }
  
  return results
}
```

## 5. LLM 响应处理

### 5.1 流式响应处理

**位置**: `src/services/api/claude.ts`

```typescript
async function* streamCompletion(
  messages: Message[],
  options: RequestOptions
): AsyncGenerator<CompletionChunk> {
  const response = await fetch(API_URL, {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${getApiKey()}`,
    },
    body: JSON.stringify({
      model: options.model,
      messages: formatMessages(messages),
      system: options.systemPrompt,
      max_tokens: 8192,
      stream: true,
    }),
  })
  
  const reader = response.body.getReader()
  const decoder = new TextDecoder()
  
  while (true) {
    const { done, value } = await reader.read()
    if (done) break
    
    const chunk = decoder.decode(value)
    
    // 解析 SSE 格式
    for (const line of chunk.split('\n')) {
      if (line.startsWith('data: ')) {
        const data = JSON.parse(line.slice(6))
        yield parseStreamEvent(data)
      }
    }
  }
}

function parseStreamEvent(event: StreamEvent): CompletionChunk {
  switch (event.type) {
    case 'content_block_start':
      return { type: 'content_start', content: event.content_block }
    case 'content_block_delta':
      return { type: 'content_delta', delta: event.delta }
    case 'content_block_stop':
      return { type: 'content_stop' }
    case 'message_delta':
      return { type: 'message_delta', delta: event.delta }
    case 'message_stop':
      return { type: 'message_stop' }
  }
}
```

## 7. 异常处理路径

### 7.1 错误分类与处理

```typescript
// src/services/api/errors.ts

function handleError(error: unknown, context: string): ErrorAction {
  if (error instanceof APIError) {
    return handleAPIError(error, context)
  }
  
  if (error instanceof PermissionError) {
    return handlePermissionError(error)
  }
  
  if (error instanceof TimeoutError) {
    return handleTimeoutError(error)
  }
  
  // 默认：记录错误并抛出
  logError(error, context)
  throw error
}

function categorizeRetryableAPIError(error: APIError): RetryDecision {
  // 5xx 错误：重试
  if (error.status >= 500 && error.status < 600) {
    return { retry: true, delay: calculateBackoff() }
  }
  
  // 429 速率限制：重试
  if (error.status === 429) {
    return { retry: true, delay: parseRetryAfter(error.headers) }
  }
  
  // 4xx 客户端错误：不重试
  return { retry: false }
}
```

### 7.2 用户友好错误消息

```typescript
// src/utils/errors.ts

function getUserFriendlyError(error: unknown): string {
  if (error instanceof APIError) {
    switch (error.status) {
      case 401:
        return 'Authentication failed. Please run /login to authenticate.'
      case 403:
        return 'Permission denied. Check your subscription status.'
      case 429:
        return 'Rate limit exceeded. Please wait a moment and try again.'
      case 500:
        return 'Server error. Please try again later.'
    }
  }
  
  if (error instanceof NetworkError) {
    return 'Network error. Please check your internet connection.'
  }
  
  return `An unexpected error occurred: ${errorMessage(error)}`
}
```

## 总结

用户输入处理流程展示了 Claude Code 的核心架构：

1. **CLI 解析**: Commander.js 处理命令行参数，区分交互/非交互模式
2. **命令路由**: 识别 slash 命令 vs 自然语言，分发到不同处理器
3. **QueryEngine 循环**: 构建提示→LLM 调用→Tool Call 检测→工具执行→结果聚合
4. **流式响应**: SSE 流式处理，增量渲染
5. **UI 渲染**: React + Ink 终端 UI 组件树
6. **错误处理**: 分类处理 API 错误、权限错误、网络错误

这个流程是理解整个系统的基础，后续文档将深入每个子流程。

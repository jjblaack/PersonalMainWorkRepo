## 2. LLM 决策与 Tool Call 生成

### 2.1 LLM 返回 Tool Call

当 LLM 决定使用工具时，返回的响应包含 `tool_use` 块：

```typescript
// API 响应结构

interface AssistantMessage {
  role: 'assistant'
  content: Array<
    | { type: 'text'; text: string }
    | ToolUseBlock
  >
  stop_reason: 'end_turn' | 'tool_use' | 'max_tokens'
}

interface ToolUseBlock {
  type: 'tool_use'
  id: string           // tool_use_id
  name: string         // 工具名称，如 'FileRead'
  input: Record<string, unknown>  // 工具输入参数
}

// 示例响应
{
  "role": "assistant",
  "content": [
    {
      "type": "text",
      "text": "我来帮你读取这个文件。"
    },
    {
      "type": "tool_use",
      "id": "toolu_abc123",
      "name": "FileRead",
      "input": {
        "file_path": "/path/to/file.txt"
      }
    }
  ],
  "stop_reason": "tool_use"
}
```

### 2.2 Tool Call 解析

**位置**: `src/QueryEngine.ts` (推断)

```typescript
function extractToolCalls(content: ContentBlockParam[]): ToolUseBlock[] {
  return content.filter((block): block is ToolUseBlock => {
    return block.type === 'tool_use'
  })
}

function processToolCalls(
  response: AssistantMessage,
  callback: (toolUse: ToolUseBlock) => Promise<void>
): void {
  const toolCalls = extractToolCalls(response.content)
  
  for (const toolCall of toolCalls) {
    // 异步处理每个工具调用
    callback(toolCall).catch(error => {
      logError(error, `Tool call ${toolCall.name}`)
    })
  }
}
```

```typescript
abstract class Tool {
  abstract name: string
  abstract description: string
  abstract inputSchema: z.ZodType
  
  // 验证输入
  validateInput(input: unknown): ValidationResult {
    const result = this.inputSchema.safeParse(input)

        valid: false,
        errors: result.error.errors.map(e => ({
          path: e.path.join('.'),
          message: e.message,
        })),
      }
    }
    
    return { valid: true, input: result.data }
  }

    input: unknown,
    context: ToolContext

}
```

## 4. 权限检查机制

### 4.1 useCanUseTool Hook

**位置**: `src/hooks/useCanUseTool.js`

```typescript
// src/hooks/useCanUseTool.js (推断)

export function useCanUseTool(): CanUseToolFn {
  const { permissionMode } = usePermissionContext()
  const { approvalHistory } = useApprovalHistory()
  
  const canUseTool = useCallback(async (
    tool: Tool,
    input: unknown,
    context: ToolContext,
    assistantMessage: AssistantMessage,
    toolUseID: string,
    forceDecision = false
  ): Promise<PermissionResult> => {
    
    // 1. 检查 bypass 模式
    if (permissionMode === 'bypassPermissions') {
      return { behavior: 'allow' }
    }
    
    // 2. 检查总是允许
    if (approvalHistory[tool.name]?.alwaysAllow) {
      return { behavior: 'allow' }
    }
    
    // 3. 检查总是拒绝
    if (approvalHistory[tool.name]?.alwaysDeny) {
      return { behavior: 'deny', reason: 'Previously denied' }
    }
    
    // 4. 检查工具权限级别
    const permissionSpec = tool.getPermission?.(input, context) 
      || getDefaultPermission(tool)
    
    // auto 模式下低风险工具自动执行
    if (permissionMode === 'auto' && permissionSpec.level === 'low') {
      return { behavior: 'allow' }
    }
    
    // 5. 需要用户审批
    if (forceDecision || permissionSpec.requiresApproval) {
      const decision = await showPermissionDialog({
        toolName: tool.name,
        toolDescription: tool.description,
        inputPreview: JSON.stringify(input, null, 2).slice(0, 500),
        riskLevel: permissionSpec.level,
      })
      
      if (decision === 'approve') {
        return { behavior: 'allow' }
      } else if (decision === 'always_allow') {
        await recordApproval(tool.name, 'always')
        return { behavior: 'allow' }
      } else {
        await recordDenial(tool.name, input)
        return { behavior: 'deny', reason: 'User denied' }
      }
    }
    
    // 6. 默认允许
    return { behavior: 'allow' }
  }, [permissionMode, approvalHistory])
  
  return canUseTool
}
```

```typescript
// QueryEngine.ts 工具执行逻辑 (推断)

private async executeToolCalls(
  toolCalls: ToolUseBlock[],
  canUseTool: CanUseToolFn,
  tools: Tool[],
  context: ToolContext
): Promise<ToolResultMessage[]> {
  const results: ToolResultMessage[] = []
  
  for (const toolCall of toolCalls) {
    // 1. 查找工具
    const tool = findTool(toolCall.name, tools)
    
    if (!tool) {
      results.push({
        role: 'tool_result',
        tool_use_id: toolCall.id,
        content: [{ 
          type: 'text', 
          text: `Unknown tool: ${toolCall.name}` 
        }],
        is_error: true,
      })
      continue
    }
    
    // 2. 验证输入
    const validation = tool.validateInput(toolCall.input)
    
    if (!validation.valid) {
      results.push({
        role: 'tool_result',
        tool_use_id: toolCall.id,
        content: [{ 
          type: 'text', 
          text: `Invalid input: ${validation.errors.map(e => e.message).join(', ')}` 
        }],
        is_error: true,
      })
      continue
    }
    
    // 3. 权限检查
    const permissionResult = await canUseTool(
      tool,
      validation.input,
      context,
      this.currentAssistantMessage,
      toolCall.id
    )
    
    if (permissionResult.behavior === 'deny') {
      results.push({
        role: 'tool_result',
        tool_use_id: toolCall.id,
        content: [{ 
          type: 'text', 
          text: `Permission denied: ${permissionResult.reason}` 
        }],
        is_error: true,
      })
      continue
    }
    
    // 4. 执行工具
    try {
      const toolResult = await tool.execute(validation.input, context)
      
      results.push({
        role: 'tool_result',
        tool_use_id: toolCall.id,
        content: toolResult.content,
        is_error: toolResult.success === false,
      })
    } catch (error) {
      results.push({
        role: 'tool_result',
        tool_use_id: toolCall.id,
        content: [{ 
          type: 'text', 
          text: `Error executing ${toolCall.name}: ${errorMessage(error)}` 
        }],
        is_error: true,
      })
    }
  }
  
  return results
}
```

## 7. 错误处理与重试

### 7.1 工具错误分类

```typescript
// src/utils/errors.ts (推断)

enum ToolErrorType {
  NOT_FOUND = 'tool_not_found',      // 工具不存在
  INVALID_INPUT = 'invalid_input',   // 输入验证失败
  PERMISSION_DENIED = 'permission_denied',
  EXECUTION_ERROR = 'execution_error',
  TIMEOUT = 'timeout',
  RESOURCE_NOT_FOUND = 'resource_not_found',
}

function classifyToolError(error: unknown, toolName: string): ToolErrorType {
  if (error instanceof NotFoundError) {
    return ToolErrorType.NOT_FOUND
  }
  
  if (error instanceof ValidationError) {
    return ToolErrorType.INVALID_INPUT
  }
  
  if (error instanceof PermissionError) {
    return ToolErrorType.PERMISSION_DENIED
  }
  
  if (error instanceof TimeoutError) {
    return ToolErrorType.TIMEOUT
  }
  
  return ToolErrorType.EXECUTION_ERROR
}
```

### 7.2 重试逻辑

```typescript
// 工具执行重试 (推断)

async function executeToolWithRetry(
  tool: Tool,
  input: unknown,
  context: ToolContext,
  options?: { maxRetries?: number }
): Promise<ToolResult> {
  const maxRetries = options?.maxRetries ?? 3
  let lastError: unknown = null
  
  for (let attempt = 1; attempt <= maxRetries; attempt++) {
    try {
      return await tool.execute(input, context)
    } catch (error) {
      lastError = error
      
      // 判断是否可重试
      if (!isRetryableError(error)) {
        throw error
      }
      
      // 计算退避延迟 (指数退避 + 抖动)
      const delay = Math.min(1000 * Math.pow(2, attempt) + Math.random() * 500, 5000)
      await sleep(delay)
    }
  }
  
  throw lastError
}

function isRetryableError(error: unknown): boolean {
  // 网络错误可重试
  if (error instanceof NetworkError) return true
  
  // 超时错误可重试
  if (error instanceof TimeoutError) return true
  
  // 5xx API 错误可重试
  if (error instanceof APIError && error.status >= 500) return true
  
  return false
}
```

## 总结

工具调用执行链是 Claude Code 与外部世界交互的核心路径：

1. **Tool Call 解析**: 从 LLM 响应中提取工具调用请求
2. **工具查找**: 在注册表中定位工具实例
3. **权限检查**: 多层次权限验证，确保操作安全
4. **工具执行**: 调用具体工具实现，执行实际操作
5. **结果返回**: 封装结果返回给 LLM，继续对话循环

这个链条中的每个环节都有错误处理和边界检查，确保系统的健壮性和安全性。

```typescript
import { Client } from '@modelcontextprotocol/sdk/client/index.js';
import { StdioClientTransport } from '@modelcontextprotocol/sdk/client/stdio.js';

class McpClient {
  constructor() {
    this.client = null;
    this.connections = new Map();
  }
  
  async connect(serverConfig) {
    const transport = new StdioClientTransport({
      command: serverConfig.command,
      args: serverConfig.args,
      env: serverConfig.env,
    });
    
    this.client = new Client({
      name: 'claude-code',
      version: getVersion(),
    }, {
      capabilities: {
        tools: {},
        resources: {},
      },
    });
    
    await this.client.connect(transport);
    return this.client;
  }
  
  async listTools() {
    const response = await this.client.listTools();
    return response.tools;
  }
  
  async callTool(name, args) {
    return this.client.callTool({ name, arguments: args });
  }
  
  async listResources() {
    const response = await this.client.listResources();
    return response.resources;
  }
  
  async readResource(uri) {
    return this.client.readResource({ uri });
  }
  
  async disconnect() {
    await this.client.close();
  }
}
```

## 4. OAuth 认证服务

### 4.1 OAuth 客户端

**位置**: `src/services/oauth/client.js`

**核心功能**:

```typescript
// src/services/oauth/client.js (推断)

import { getOauthConfig } from '../../constants/oauth.js';
import { setKeychainItem, getKeychainItem } from '../../utils/secureStorage/keychain.js';

class OAuthClient {
  async startAuthFlow() {
    const config = await getOauthConfig();
    const state = generateRandomState();
    const codeVerifier = generateCodeVerifier();
    
    // 存储 PKCE 参数
    await setKeychainItem('claude-code', 'oauth_state', state);
    await setKeychainItem('claude-code', 'code_verifier', codeVerifier);
    
    // 构建授权 URL
    const authUrl = this.buildAuthUrl(config, state);
    
    // 在浏览器中打开
    await open(authUrl);
    
    // 启动本地服务器接收回调
    return this.waitForCallback();
  }
  
  async exchangeCodeForToken(code, receivedState) {
    const storedState = await getKeychainItem('claude-code', 'oauth_state');
    
    if (receivedState !== storedState) {
      throw new Error('State mismatch');
    }
    
    const codeVerifier = await getKeychainItem('claude-code', 'code_verifier');
    
    const response = await fetch(config.tokenEndpoint, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        grant_type: 'authorization_code',
        code,
        redirect_uri: 'http://localhost:8989/callback',
        code_verifier: codeVerifier,
      }),
    });
    
    const tokens = await response.json();
    
    // 存储 token
    await setKeychainItem('claude-code', 'access_token', tokens.access_token);
    await setKeychainItem('claude-code', 'refresh_token', tokens.refresh_token);
    
    return tokens;
  }
  
  async refreshAccessToken() {
    const refreshToken = await getKeychainItem('claude-code', 'refresh_token');
    
    const response = await fetch(config.tokenEndpoint, {
      method: 'POST',
      body: JSON.stringify({
        grant_type: 'refresh_token',
        refresh_token: refreshToken,
      }),
    });
    
    return response.json();
  }
}
```

### 4.2 认证状态检查

```typescript
// src/services/oauth/client.js (推断)

export async function populateOAuthAccountInfo() {
  const accessToken = await getKeychainItem('claude-code', 'access_token');
  
  if (!accessToken) {
    return null;
  }
  
  // 检查 token 是否过期
  const expiresAt = await getKeychainItem('claude-code', 'token_expires_at');
  
  if (Date.now() > expiresAt) {
    // Token 过期，刷新
    const newTokens = await refreshAccessToken();
    return newTokens;
  }
  
  return { accessToken };
}
```

```typescript
import { LanguageClient } from 'vscode-languageclient/node.js';

class LspManager {
  constructor() {
    this.clients = new Map();
  }
  
  async startServer(rootPath, serverOptions) {
    const client = new LanguageClient(
      `lsp-${path.basename(rootPath)}`,
      serverOptions,
      {
        documentSelector: [
          { scheme: 'file', language: 'typescript' },
          { scheme: 'file', language: 'javascript' },
          { scheme: 'file', language: 'python' },
        ],
      }
    );
    
    await client.start();
    this.clients.set(rootPath, client);
    
    return client;
  }
  
  async getCompletion(uri, position) {
    const client = this.getClientForUri(uri);
    if (!client) return null;
    
    return client.sendRequest('textDocument/completion', {
      textDocument: { uri },
      position,
    });
  }
  
  async getDiagnostics(uri) {
    // 获取诊断信息
  }
  
  async shutdown() {
    for (const client of this.clients.values()) {
      await client.stop();
    }
    this.clients.clear();
  }
}
```

```typescript
// src/services/tokenEstimation.ts (推断)

import { encoding_for_model } from '@anthropic-ai/tokenizer';

export function estimateTokens(text: string, model: string): number {
  const encoder = encoding_for_model(model as any);
  const tokens = encoder.encode(text);
  encoder.free();
  return tokens.length;
}

export function estimateMessageTokens(message: Message): number {
  let tokens = 0;
  
  if (message.content) {
    if (typeof message.content === 'string') {
      tokens += estimateTokens(message.content, 'claude-sonnet-4-20250514');
    } else if (Array.isArray(message.content)) {
      for (const block of message.content) {
        if (block.type === 'text') {
          tokens += estimateTokens(block.text, 'claude-sonnet-4-20250514');
        } else if (block.type === 'image') {
          // 图片 token 估算
          tokens += estimateImageTokens(block.source);
        }
      }
    }
  }
  
  return tokens;
}
```

## 总结

服务层是 Claude Code 与外部世界交互的桥梁，提供了：

1. **API 封装**: 简化外部 API 调用
2. **协议支持**: MCP、LSP、OAuth 等

服务层的设计使得核心业务逻辑与外部系统解耦，便于维护和扩展。

## 2. OAuth 2.0 认证流程

### 2.1 OAuthService 类结构

**位置**: `src/services/oauth/index.ts`

```typescript
export class OAuthService {
  private codeVerifier: string
  private authCodeListener: AuthCodeListener | null = null
  private port: number | null = null
  private manualAuthCodeResolver: ((authorizationCode: string) => void) | null = null

  constructor() {
    // 生成 PKCE code verifier
    this.codeVerifier = crypto.generateCodeVerifier()
  }

  async startOAuthFlow(
    authURLHandler: (url: string, automaticUrl?: string) => Promise<void>,
    options?: {
      loginWithClaudeAi?: boolean
      inferenceOnly?: boolean
      expiresIn?: number
      orgUUID?: string
      loginHint?: string
      loginMethod?: string
      skipBrowserOpen?: boolean
    },
  ): Promise<OAuthTokens> {
    // === 阶段 1: 准备授权 ===
    
    // 1.1 创建授权码监听器
    this.authCodeListener = new AuthCodeListener()
    this.port = await this.authCodeListener.start()
    
    // 1.2 生成 PKCE 挑战
    const codeChallenge = crypto.generateCodeChallenge(this.codeVerifier)
    const state = crypto.generateState()
    
    // 1.3 构建授权 URL
    const opts = {
      codeChallenge,
      state,
      port: this.port,
      loginWithClaudeAi: options?.loginWithClaudeAi,
      inferenceOnly: options?.inferenceOnly,
    }
    
    const manualFlowUrl = client.buildAuthUrl({ ...opts, isManual: true })
    const automaticFlowUrl = client.buildAuthUrl({ ...opts, isManual: false })
    
    // === 阶段 2: 获取授权码 ===
    
    // 2.1 等待授权码 (自动或手动)
    const authorizationCode = await this.waitForAuthorizationCode(
      state,
      async () => {
        if (options?.skipBrowserOpen) {
          await authURLHandler(manualFlowUrl, automaticFlowUrl)
        } else {
          await authURLHandler(manualFlowUrl)
          await openBrowser(automaticFlowUrl)
        }
      },
    )
    
    // === 阶段 3: 交换令牌 ===
    
    // 3.1 用授权码交换访问令牌
    const tokenResponse = await client.exchangeCodeForTokens(
      authorizationCode,
      state,
      this.codeVerifier,
      this.port!,
    )
    
    // === 阶段 4: 获取用户信息 ===
    
    // 4.1 获取用户 Profile 信息
    const profileInfo = await client.fetchProfileInfo(
      tokenResponse.access_token,
    )
    
    // === 阶段 5: 返回令牌 ===
    
    return this.formatTokens(
      tokenResponse,
      profileInfo.subscriptionType,
      profileInfo.rateLimitTier,
      profileInfo.rawProfile,
    )
  }
}
```

### 2.2 授权码监听器

**位置**: `src/services/oauth/auth-code-listener.ts`

```typescript
export class AuthCodeListener {
  private server: http.Server | null = null
  private port: number
  private authorizationCode: string | null = null
  private receivedState: string | null = null

  async start(): Promise<number> {
    return new Promise((resolve, reject) => {
      this.server = http.createServer(async (req, res) => {
        const url = new URL(req.url!, `http://localhost:${this.port}`)
        
        // 仅处理 /callback 路径
        if (url.pathname !== '/callback') {
          res.writeHead(404)
          res.end()
          return
        }
        
        // 提取授权码和 state
        const code = url.searchParams.get('code')
        const state = url.searchParams.get('state')
        const error = url.searchParams.get('error')
        
        if (error) {
          res.writeHead(400)
          res.end(`OAuth error: ${error}`)
          reject(new Error(error))
          return
        }
        
        if (!code || !state) {
          res.writeHead(400)
          res.end('Missing code or state')
          reject(new Error('Missing code or state'))
          return
        }
        
        // 保存授权码
        this.authorizationCode = code
        this.receivedState = state
        
        // 返回成功页面
        res.writeHead(200, { 'Content-Type': 'text/html' })
        res.end(this.buildSuccessHtml())
      })
      
      // 监听随机端口
      this.server.listen(0, () => {
        const address = this.server!.address() as http.AddressInfo
        this.port = address.port
        resolve(this.port)
      })
    })
  }

  async waitForAuthorizationCode(
    expectedState: string,
    onReady: () => Promise<void>,
  ): Promise<string> {
    // 启动授权页面
    await onReady()
    
    // 轮询等待授权码
    while (!this.authorizationCode) {
      await sleep(100)
    }
    
    // 验证 state
    if (this.receivedState !== expectedState) {
      throw new Error('State mismatch')
    }
    
    return this.authorizationCode!
  }
}
```

### 2.3 构建授权 URL

**位置**: `src/services/oauth/client.ts:46-105`

```typescript
export function buildAuthUrl({
  codeChallenge,
  state,
  port,
  isManual,
  loginWithClaudeAi,
  inferenceOnly,
}: {
  codeChallenge: string
  state: string
  port: number
  isManual: boolean
  loginWithClaudeAi?: boolean
  inferenceOnly?: boolean
}): string {
  // 选择授权 URL 基础
  const authUrlBase = loginWithClaudeAi
    ? getOauthConfig().CLAUDE_AI_AUTHORIZE_URL
    : getOauthConfig().CONSOLE_AUTHORIZE_URL

  const authUrl = new URL(authUrlBase)
  
  // OAuth 2.0 必需参数
  authUrl.searchParams.append('client_id', getOauthConfig().CLIENT_ID)
  authUrl.searchParams.append('response_type', 'code')
  
  // 重定向 URL
  authUrl.searchParams.append(
    'redirect_uri',
    isManual
      ? getOauthConfig().MANUAL_REDIRECT_URL
      : `http://localhost:${port}/callback`,
  )
  
  // 授权范围
  const scopesToUse = inferenceOnly
    ? [CLAUDE_AI_INFERENCE_SCOPE]  // 仅推理权限
    : ALL_OAUTH_SCOPES
  authUrl.searchParams.append('scope', scopesToUse.join(' '))
  
  // PKCE 参数
  authUrl.searchParams.append('code_challenge', codeChallenge)
  authUrl.searchParams.append('code_challenge_method', 'S256')
  
  // State 参数 (防 CSRF)
  authUrl.searchParams.append('state', state)
  
  return authUrl.toString()
}
```

## 3. PKCE 安全机制

### 3.1 PKCE 流程

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                          PKCE 流程                                           │
└─────────────────────────────────────────────────────────────────────────────┘

客户端                          授权服务器
  │                                 │
  │  1. 生成 code_verifier          │
  │     (随机字符串)                 │
  │                                 │
  │  2. 计算 code_challenge         │
  │     = SHA256(code_verifier)     │
  │                                 │
  │  3. 授权请求                     │
  │     + code_challenge           │
  │     + code_challenge_method    │
  │────────────────────────────────>│
  │                                 │
  │  4. 用户登录并授权               │
  │                                 │
  │  5. 返回 authorization_code     │
  │<────────────────────────────────│
  │                                 │
  │  6. 令牌请求                     │
  │     + code_verifier            │
  │────────────────────────────────>│
  │                                 │
  │  7. 验证 code_verifier          │
  │     SHA256( verifier) == challenge
  │                                 │
  │  8. 返回 access_token           │
  │<────────────────────────────────│
```

### 3.2 PKCE 加密工具

**位置**: `src/services/oauth/crypto.ts`

```typescript
// 生成 code verifier (随机字符串)

export function generateCodeVerifier(): string {
  // 生成 32 字节的随机数据
  const buffer = crypto.getRandomValues(new Uint8Array(32))
  
  // Base64URL 编码
  return base64UrlEncode(buffer)
}

// 生成 code challenge

export function generateCodeChallenge(verifier: string): string {
  // SHA256 哈希
  const encoder = new TextEncoder()
  const data = encoder.encode(verifier)
  
  // 计算哈希
  const hash = crypto.subtle.digest('SHA-256', data)
  
  // Base64URL 编码
  return base64UrlEncode(new Uint8Array(hash))
}

// Base64URL 编码

function base64UrlEncode(buffer: Uint8Array): string {
  let binary = ''
  for (let i = 0; i < buffer.byteLength; i++) {
    binary += String.fromCharCode(buffer[i])
  }
  
  // 标准 Base64 转 Base64URL
  return btoa(binary)
    .replace(/\+/g, '-')
    .replace(/\//g, '_')
    .replace(/=+$/, '')
}

// 生成 state (防 CSRF)

export function generateState(): string {
  const buffer = crypto.getRandomValues(new Uint8Array(16))
  return base64UrlEncode(buffer)
}
```

```typescript
export async function exchangeCodeForTokens(
  authorizationCode: string,
  state: string,
  codeVerifier: string,
  port: number,
  useManualRedirect: boolean = false,
  expiresIn?: number,
): Promise<OAuthTokenExchangeResponse> {
  const requestBody: Record<string, string | number> = {
    grant_type: 'authorization_code',
    code: authorizationCode,
    redirect_uri: useManualRedirect
      ? getOauthConfig().MANUAL_REDIRECT_URL
      : `http://localhost:${port}/callback`,
    client_id: getOauthConfig().CLIENT_ID,
    code_verifier: codeVerifier,  // PKCE verifier
    state,
  }
  
  if (expiresIn !== undefined) {
    requestBody.expires_in = expiresIn
  }
  
  const response = await axios.post(getOauthConfig().TOKEN_URL, requestBody, {
    headers: { 'Content-Type': 'application/json' },
    timeout: 15000,
  })
  
  if (response.status !== 200) {
    throw new Error(
      response.status === 401
        ? 'Authentication failed: Invalid authorization code'
        : `Token exchange failed (${response.status})`
    )
  }
  
  return response.data
}

// 响应结构

interface OAuthTokenExchangeResponse {
  access_token: string
  refresh_token: string
  expires_in: number           // 秒
  token_type: 'Bearer'
  scope: string
  account?: {
    uuid: string
    email_address: string
  }
  organization?: {
    uuid: string
    name: string
  }
}
```

```typescript
export async function refreshOAuthToken(
  refreshToken: string,
  { scopes }: { scopes?: string[] } = {},
): Promise<OAuthTokens> {
  const requestBody = {
    grant_type: 'refresh_token',
    refresh_token: refreshToken,
    client_id: getOauthConfig().CLIENT_ID,
    // 请求特定的 scopes
    scope: (scopes?.length ? scopes : CLAUDE_AI_OAUTH_SCOPES).join(' '),
  }
  
  try {
    const response = await axios.post(getOauthConfig().TOKEN_URL, requestBody, {
      headers: { 'Content-Type': 'application/json' },
      timeout: 15000,
    })
    
    if (response.status !== 200) {
      throw new Error(`Token refresh failed: ${response.statusText}`)
    }
    
    const data = response.data as OAuthTokenExchangeResponse
    const {
      access_token: accessToken,
      refresh_token: newRefreshToken = refreshToken,
      expires_in: expiresIn,
    } = data
    
    const expiresAt = Date.now() + expiresIn * 1000
    const scopes = parseScopes(data.scope)
    
    // 获取用户 Profile 信息
    const profileInfo = await fetchProfileInfo(accessToken)
    
    return formatTokens(data, profileInfo)
  } catch (error) {
    // 刷新失败，需要重新登录
    throw new OAuthRefreshError(error)
  }
}
```

```typescript
async function checkAndRefreshOAuthTokenIfNeeded(): Promise<boolean> {
  const config = getGlobalConfig()
  const tokens = await getOAuthTokensFromSecureStorage()
  
  if (!tokens) {
    return false
  }
  
  // 检查是否即将过期 (提前 5 分钟刷新)
  const expiresInMs = tokens.expiresAt - Date.now()
  const refreshThreshold = 5 * 60 * 1000  // 5 分钟
  
  if (expiresInMs < refreshThreshold) {
    try {
      // 刷新 token
      const newTokens = await refreshOAuthToken(tokens.refreshToken)
      
      // 保存新 token
      await saveOAuthTokens(newTokens)
      
      return true
    } catch (error) {
      // 刷新失败，需要重新登录
      logError('Token refresh failed', error)
      return false
    }
  }
  
  return true
}
```

```typescript
async function createSession(user: User): Promise<Session> {
  const sessionId = generateSessionId()
  
  // 创建 JWT
  const token = await new SignJWT({
    user_id: user.id,
    email: user.email,
    org_uuid: user.orgUuid,
  })
    .setProtectedHeader({ alg: 'HS256' })
    .setIssuedAt()
    .setExpirationTime('24h')
    .setJti(sessionId)
    .sign(getSecret())
  
  // 保存会话
  await saveSessionToStorage(sessionId, {
    userId: user.id,
    createdAt: Date.now(),
    expiresAt: Date.now() + 24 * 60 * 60 * 1000,
  })
  
  return {
    id: sessionId,
    token,
    user,
  }
}
```

```typescript
// src/middleware/auth.ts (推断)

async function authMiddleware(
  req: Request,
  res: Response,
  next: NextFunction,
): Promise<void> {
  const authHeader = req.headers.authorization
  
  if (!authHeader || !authHeader.startsWith('Bearer ')) {
    res.status(401).json({ error: 'Missing authorization header' })
    return
  }
  
  const token = authHeader.slice(7)
  
  try {
    // 验证 JWT
    const payload = await verifySessionToken(token)
    
    // 检查是否过期
    if (payload.exp < Date.now() / 1000) {
      throw new Error('Token expired')
    }
    
    // 附加用户信息到请求
    req.user = {
      id: payload.sub,
      email: payload.email,
      orgUuid: payload.org_uuid,
    }
    
    next()
  } catch (error) {
    res.status(401).json({ error: 'Invalid or expired token' })
  }
}
```

### 7.2 场景：Token 过期自动刷新

```
用户运行：claude

1. 从 Keychain 加载 OAuth Token

2. 检查过期时间：
   expiresAt = 1712000000000  (已过期或即将过期)
   now = 1711999900000

3. expiresAt - now < 5 分钟 → 需要刷新

4. 调用 refreshOAuthToken(refreshToken)

5. 刷新成功：
   - 获取新的 access_token
   - 获取新的 refresh_token
   - 更新 Keychain 存储

6. 使用新 token 继续

刷新失败处理：
- 如果刷新失败 (refresh_token 过期或被撤销)
- 清除本地存储
- 提示用户重新登录
```

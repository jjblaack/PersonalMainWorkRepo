## 1. 组件系统概述

### 1.1 架构设计原则

Claude Code 的组件系统遵循以下设计原则：

| 原则 | 说明 | 实现方式 |
|------|------|----------|
| **单一职责** | 每个组件只负责一个功能 | 细粒度组件拆分 |
| **组合优于继承** | 通过组合构建复杂 UI | HOC 和 Render Props |
| **关注点分离** | UI 与逻辑分离 | 自定义 Hooks |
| **类型安全** | 完整的 TypeScript 类型 | 严格的接口定义 |
| **性能优先** | 避免不必要的渲染 | React.memo + 选择器 |

```typescript
// 主题定义

type ThemeMode = 'light' | 'dark' | 'auto';

interface Theme {
  colors: {
    background: string;
    foreground: string;
    primary: string;
    secondary: string;
    success: string;
    warning: string;
    error: string;
    muted: string;
  };
  spacing: {
    xs: number;
    sm: number;
    md: number;
    lg: number;
    xl: number;
  };
  typography: {
    fontFamily: string;
    fontSize: {
      xs: number;
      sm: number;
      base: number;
      lg: number;
      xl: number;
    };
  };
}

// 主题提供器实现

export function ThemeProvider({ children }: ThemeProviderProps) {
  const [themeMode, setThemeMode] = useSetting('themeMode', 'auto');
  const [systemTheme, setSystemTheme] = useState<'light' | 'dark'>('dark');
  
  // 监听系统主题变化
  useEffect(() => {
    if (themeMode === 'auto') {
      const mediaQuery = window.matchMedia('(prefers-color-scheme: dark)');
      const handler = (e: MediaQueryListEvent) => {
        setSystemTheme(e.matches ? 'dark' : 'light');
      };
      mediaQuery.addEventListener('change', handler);
      return () => mediaQuery.removeEventListener('change', handler);
    }
  }, [themeMode]);
  
  // 计算当前主题
  const currentTheme: ThemeMode = themeMode === 'auto' ? systemTheme : themeMode;
  
  const theme: Theme = useMemo(() => ({
    colors: currentTheme === 'dark' ? darkColors : lightColors,
    spacing,
    typography,
  }), [currentTheme]);
  
  return (
    <ThemeContext.Provider value={{ theme, themeMode, setThemeMode }}>
      {children}
    </ThemeContext.Provider>
  );
}
```

```typescript
export interface ThemedBoxProps {
  variant?: 'default' | 'primary' | 'secondary' | 'danger' | 'success';
  padding?: keyof typeof spacing;
  margin?: keyof typeof spacing;
  borderRadius?: 'none' | 'sm' | 'md' | 'lg' | 'full';
  children: React.ReactNode;
}

export function ThemedBox({
  variant = 'default',
  padding,
  margin,
  borderRadius = 'md',
  children
}: ThemedBoxProps) {
  const { theme } = useTheme();
  
  const backgroundColor = useMemo(() => {
    switch (variant) {
      case 'primary': return theme.colors.primary;
      case 'secondary': return theme.colors.secondary;
      case 'danger': return theme.colors.error;
      case 'success': return theme.colors.success;
      default: return theme.colors.background;
    }
  }, [variant, theme]);
  
  return (
    <Box
      backgroundColor={backgroundColor}
      padding={padding ? spacing[padding] : undefined}
      margin={margin ? spacing[margin] : undefined}
      borderRadius={borderRadiusValues[borderRadius]}
    >
      {children}
    </Box>
  );
}
```

#### 3.2.3 Button

```typescript
// src/components/design-system/Button.tsx

export interface ButtonProps {
  variant?: 'primary' | 'secondary' | 'danger' | 'ghost';
  size?: 'sm' | 'md' | 'lg';
  loading?: boolean;
  disabled?: boolean;
  leftIcon?: React.ReactNode;
  rightIcon?: React.ReactNode;
  onClick?: () => void;
  children: React.ReactNode;
}

export function Button({
  variant = 'primary',
  size = 'md',
  loading = false,
  disabled = false,
  leftIcon,
  rightIcon,
  onClick,
  children
}: ButtonProps) {
  const { theme } = useTheme();
  
  const styles = useMemo(() => {
    const baseStyle = {
      padding: size === 'sm' ? [spacing.xs, spacing.sm] : [spacing.sm, spacing.md],
      borderRadius: 6,
      cursor: disabled || loading ? 'default' : 'pointer',
    };
    
    switch (variant) {
      case 'primary':
        return {
          ...baseStyle,
          backgroundColor: theme.colors.primary,
          color: theme.colors.background,
        };
      case 'secondary':
        return {
          ...baseStyle,
          backgroundColor: theme.colors.secondary,
          color: theme.colors.foreground,
        };
      case 'danger':
        return {
          ...baseStyle,
          backgroundColor: theme.colors.error,
          color: theme.colors.background,
        };
      case 'ghost':
        return {
          ...baseStyle,
          backgroundColor: 'transparent',
          color: theme.colors.foreground,
        };
    }
  }, [variant, size, disabled, loading, theme]);
  
  return (
    <Box style={styles} onClick={disabled || loading ? undefined : onClick}>
      {loading ? <Spinner size="sm" /> : leftIcon}
      <Text>{children}</Text>
      {!loading && rightIcon}
    </Box>
  );
}
```

```typescript
export interface DialogProps {
  isOpen: boolean;
  onClose: () => void;
  title: string;
  description?: string;
  size?: 'sm' | 'md' | 'lg' | 'xl';
  showCloseButton?: boolean;
  children: React.ReactNode;
  footer?: React.ReactNode;
}

export function Dialog({
  isOpen,
  onClose,
  title,
  description,
  size = 'md',
  showCloseButton = true,
  children,
  footer
}: DialogProps) {
  useInput((input, key) => {
    if (key.escape && isOpen) {
      onClose();
    }
  }, { isActive: isOpen });
  
  if (!isOpen) return null;
  
  const width = { sm: 40, md: 60, lg: 80, xl: 100 }[size];
  
  return (
    <Box
      position="absolute"
      top={0}
      left={0}
      right={0}
      bottom={0}
      backgroundColor="rgba(0,0,0,0.5)"
      justifyContent="center"
      alignItems="center"
    >
      <Box
        width={width}
        backgroundColor="background"
        borderRadius="lg"
        border="single"
        flexDirection="column"
      >
        {/* Header */}
        <Box
          padding={1}
          borderBottom="single"
          justifyContent="space-between"
        >
          <Box>
            <Text bold>{title}</Text>
            {description && (
              <Text dimColor marginTop={1}>{description}</Text>
            )}
          </Box>
          {showCloseButton && (
            <Text button onClick={onClose} dimColor>[×]</Text>
          )}
        </Box>
        
        {/* Content */}
        <Box padding={1} flexGrow={1}>
          {children}
        </Box>
        
        {/* Footer */}
        {footer && (
          <Box
            padding={1}
            borderTop="single"
            justifyContent="flex-end"
          >
            {footer}
          </Box>
        )}
      </Box>
    </Box>
  );
}
```

```typescript
// src/components/design-system/ProgressBar.tsx

export interface ProgressBarProps {
  value: number;  // 0-1
  width?: number;
  showLabel?: boolean;
  variant?: 'default' | 'success' | 'warning' | 'danger';
}

export function ProgressBar({
  value,
  width = 40,
  showLabel = true,
  variant = 'default'
}: ProgressBarProps) {
  const { theme } = useTheme();
  
  const filledWidth = Math.floor(width * value);
  const emptyWidth = width - filledWidth;
  
  const color = {
    default: 'cyan',
    success: 'green',
    warning: 'yellow',
    danger: 'red',
  }[variant];
  
  return (
    <Box flexDirection="row" alignItems="center">
      <Text color={color}>
        {'█'.repeat(filledWidth)}
      </Text>
      <Text dimColor>
        {'░'.repeat(emptyWidth)}
      </Text>
      {showLabel && (
        <Text dimColor marginLeft={1}>
          {Math.round(value * 100)}%
        </Text>
      )}
    </Box>
  );
}
```

```typescript
function MainContent() {
  const appState = useAppState();
  const { settings } = appState;

  if (appState.isNonInteractive) {
    return <NonInteractiveLayout />;
  }
  
  if (settings.fullscreen) {
    return <FullscreenLayout />;
  }
  
  return <DefaultLayout />;
}
```

```typescript
// src/components/ContextSuggestions.tsx (简化)

export function ContextSuggestions({ suggestions }: Props) {
  const [selectedIndex, setSelectedIndex] = useState(0);
  const addContext = useAddContext();
  
  useInput((input, key) => {
    if (key.key === 'right') {
      setSelectedIndex(i => Math.min(suggestions.length - 1, i + 1));
    } else if (key.key === 'left') {
      setSelectedIndex(i => Math.max(0, i - 1));
    } else if (key.key === 'enter') {
      addContext(suggestions[selectedIndex]);
    }
  });
  
  if (suggestions.length === 0) return null;
  
  return (
    <Box flexDirection="column" borderTop="single" padding={1}>
      <Text bold dimColor>Suggested Context:</Text>
      <Box flexDirection="row" marginTop={1}>
        {suggestions.map((suggestion, index) => (
          <Box
            key={suggestion.id}
            paddingX={2}
            paddingY={1}
            border="single"
            borderColor={index === selectedIndex ? 'blue' : 'dim'}
            backgroundColor={index === selectedIndex ? 'blue' : undefined}
            marginRight={1}
          >
            <Text>{suggestion.label}</Text>
          </Box>
        ))}
      </Box>
    </Box>
  );
}
```

```typescript
// src/state/store.ts

// 简单的 Zustand 风格状态管理

type Listener<T> = (state: T) => void;

export type Store<T> = {
  getState: () => T;
  setState: (updater: (prev: T) => T) => void;
  subscribe: (listener: Listener<T>) => () => void;
};

export function createStore<T>(
  initialState: T,
  onChange?: OnChange<T>,
): Store<T> {
  let state = initialState;
  const listeners = new Set<Listener<T>>();
  
  return {
    getState: () => state,
    setState: (updater) => {
      const prev = state;
      state = updater(prev);
      listeners.forEach(listener => listener(state));
      onChange?.(state, prev);
    },
    subscribe: (listener) => {
      listeners.add(listener);
      return () => listeners.delete(listener);
    },
  };
}
```

### 9.2 useAppState Hook

```typescript
// src/components/hooks/useAppState.ts

export function useAppState<T>(
  selector: (state: AppState) => T,
  equalityFn?: (a: T, b: T) => boolean
): T {
  const store = useContext(AppStateContext);
  const [state, setState] = useState(() => selector(store.getState()));
  
  useEffect(() => {
    return store.subscribe((newState) => {
      const newValue = selector(newState);
      if (!equalityFn || !equalityFn(state, newValue)) {
        setState(newValue);
      }
    });
  }, [store, selector, equalityFn]);
  
  return state;
}
```

```typescript
export function useThrottledToolProgress(toolUse: ToolUse, interval = 100) {
  const [progress, setProgress] = useState<ToolProgress | null>(null);
  const lastUpdate = useRef(0);
  
  useEffect(() => {
    return toolUse.subscribe(update => {

        setProgress(update);
      }
```

## 11. 总结

### 架构优势

| 方面 | 实现 | 收益 |
|------|------|------|

3. **选择器模式**: 细粒度状态订阅
